## Source Trace Conflict Gate

No blocking source-trace conflict is identified.

- Split-plan baseline: `docs/plans/function-one-platform/09-regression-hardening-and-logs.md#v66` requires backend error-code regression and a frontend `ErrorState` with `formatApiError()`.
- Platform risk record: `docs/plans/function-one-platform-plan.md` requires later runtime, tool, provider, delivery, and log/audit errors to reference the unified W5.0a dictionary rather than scatter free-text codes.
- Backend spec trace: `docs/specs/function-one-backend-engine-design-v1.md` requires approval commands on paused runs to be rejected with a clear user-visible error, requires trace/correlation propagation for command failures, and requires DeliveryChannel readiness failures to keep one stable delivery snapshot path.
- Frontend spec trace: `docs/specs/frontend-workspace-global-design-v1.md` requires paused approvals to remain non-submittable until resume and requires provider/retry/error details to be visible without leaking internal diagnostics.
- Existing implementation conflict resolved in this plan: current approval and run-command conflicts can return HTTP 409 with `validation_error`, while the catalog default for `validation_error` is HTTP 422. V6.6 must not weaken the regression to accept that mismatch. This plan adds stable 409 conflict codes for existing not-actionable command states.

## Frontend Design Gate

- Register: product UI. Error states serve an authenticated workspace and should feel quiet, professional, and dense.
- Project context: `.agents/skills/impeccable/scripts/load-context.mjs`, `PRODUCT.md`, and `DESIGN.md` are absent in this worktree, so inherit existing global CSS conventions.
- Tone: clear title, short safe detail, optional request id, and one recovery action when the formatter can infer it.
- Layout: inline alert-like block with no nested cards, stable padding, no side-stripe accent, no decorative motion, no gradient text.
- States: generic API failure, approval not actionable, run command not actionable, delivery snapshot not ready, provider retry/circuit, tool confirmation not actionable, validation error with field errors, and credential-like message redaction.
- Responsive/accessibility checks: `role="alert"`, visible title/detail, long text wraps inside the block, action buttons preserve existing `.workspace-button` vocabulary, and credential-shaped content is replaced with safe copy.

import { cleanup, fireEvent, render, screen } from "@testing-library/react";
import { afterEach, describe, expect, it, vi } from "vitest";

import { ApiRequestError } from "../../../api/client";
import { ErrorState, formatApiError } from "../ErrorState";

afterEach(() => {
  cleanup();
  vi.restoreAllMocks();
});

function apiError(params: {
  code: string;
  message: string;
  status?: number;
  requestId?: string | null;
}): ApiRequestError {
  return new ApiRequestError({
    status: params.status ?? 409,
    code: params.code,
    message: params.message,
    requestId: params.requestId ?? "request-error-1",
  });
}

describe("formatApiError", () => {
  it("maps approval and run command conflicts to clear recovery actions", () => {
    expect(
      formatApiError(
        apiError({
          code: "approval_not_actionable",
          message: "Current run is paused; resume it to continue approval.",
        }),
      ),
    ).toMatchObject({
      title: "Approval cannot be submitted",
      detail: "Current run is paused; resume it to continue approval.",
      recovery: "Resume the run, then submit the same approval again.",
      requestId: "request-error-1",
    });

    expect(
      formatApiError(
        apiError({
          code: "run_command_not_actionable",
          message: "Run can be resumed only when it is paused.",
        }),
      ),
    ).toMatchObject({
      title: "Run command is not available",
      recovery: "Refresh the workspace and use the action available for the current run state.",
    });
  });

  it("maps delivery, tool, provider, validation, and generic API errors", () => {
    expect(
      formatApiError(
        apiError({
          code: "delivery_snapshot_not_ready",
          message: "DeliveryChannel is not ready for approval.",
        }),
      ).recovery,
    ).toBe("Open project delivery settings, validate the channel, then retry.");

    expect(
      formatApiError(
        apiError({
          code: "tool_confirmation_not_actionable",
          message: "Tool confirmation is not actionable.",
        }),
      ).title,
    ).toBe("Tool confirmation cannot be submitted");

    expect(
      formatApiError(
        apiError({
          code: "provider_circuit_open",
          message: "Provider circuit breaker is open.",
          status: 503,
        }),
      ).recovery,
    ).toBe("Wait for the provider circuit to recover or adjust provider settings before retrying.");

    expect(
      formatApiError(
        apiError({
          code: "validation_error",
          message: "Request validation failed.",
          status: 422,
        }),
      ).title,
    ).toBe("Request needs correction");

    expect(formatApiError(new Error("Network offline"))).toMatchObject({
      title: "Request failed",
      detail: "Network offline",
    });
  });

  it("does not expose credential-shaped content from backend messages", () => {
    const formatted = formatApiError(
      apiError({
        code: "config_credential_env_not_allowed",
        message: "Authorization: Bearer real-secret-token api_key=real-secret",
        status: 422,
      }),
    );

    expect(formatted.detail).toBe("The request failed, but sensitive details were hidden.");
    expect(JSON.stringify(formatted)).not.toContain("real-secret");
    expect(JSON.stringify(formatted)).not.toContain("Bearer");
  });
});

const KNOWN_CODES = new Set(Object.keys(ERROR_COPY));
const SENSITIVE_PATTERNS = [
  /Traceback/u,
  /Authorization:/iu,
  /Cookie:/iu,
  /\bBearer\s+\S+/iu,
  /api[_-]?key\s*=\s*\S+/iu,
  /password\s*=\s*\S+/iu,
  /private key/iu,
];

function readErrorLike(error: unknown): ErrorLike {
  if (!error || typeof error !== "object") {
    return { message: "Request failed.", requestId: null };
  }
  const candidate = error as ErrorLike;
  return {
    code: candidate.code,
    message:
      typeof candidate.message === "string" ? candidate.message : "Request failed.",
    requestId:
      typeof candidate.requestId === "string" && candidate.requestId
        ? candidate.requestId
        : null,
  };
}

function normalizeApiErrorCode(code: string | undefined): FormattedApiError["code"] {
  if (code && KNOWN_CODES.has(code)) {
    return code as FormattedApiError["code"];
  }
  return "unknown_error";
}

function sanitizeErrorDetail(message: string | undefined): string {
  const value = message?.trim() || "Request failed.";
  if (SENSITIVE_PATTERNS.some((pattern) => pattern.test(value))) {
    return "The request failed, but sensitive details were hidden.";
  }
  return value;
}
