import type { MetricSet } from "../../api/types";

const metricOrder: Array<keyof MetricSet> = [
  "duration_ms",
  "input_tokens",
  "output_tokens",
  "total_tokens",
  "attempt_index",
  "context_file_count",
  "reasoning_step_count",
  "tool_call_count",
  "changed_file_count",
  "added_line_count",
  "removed_line_count",
  "generated_test_count",
  "executed_test_count",
  "passed_test_count",
  "failed_test_count",
  "skipped_test_count",
  "test_gap_count",
  "retry_index",
  "source_attempt_index",
  "delivery_artifact_count",
];

export function MetricGrid({ metrics }: { metrics: MetricSet }): JSX.Element | null {
  const entries = hideInapplicableMetrics(metrics);
  if (entries.length === 0) {
    return null;
  }

  return (
    <div className="inspector-metric-grid" aria-label="Metrics grid">
      {entries.map(([label, value]) => (
        <div key={label} className="inspector-metric-grid__item">
          <span>{label}</span>
          <strong>{value}</strong>
        </div>
      ))}
    </div>
  );
}

export function hideInapplicableMetrics(metrics: MetricSet): Array<[string, string]> {
  return metricOrder.flatMap((key) => {
    const value = metrics[key];
    if (value === undefined || value === null) {
      return [];
    }
    return [[formatMetricLabel(key), formatMetricValue(key, value)]];
  });
}

function formatMetricLabel(value: string): string {
  return value
    .split("_")
    .map((part) => part.charAt(0).toUpperCase() + part.slice(1))
    .join(" ");
}

function formatMetricValue(key: string, value: number): string {
  if (key.endsWith("_ms")) {
    if (value >= 60000) {
      const totalSeconds = Math.round(value / 1000);
      const minutes = Math.floor(totalSeconds / 60);
      const seconds = totalSeconds % 60;
      return seconds === 0 ? `${minutes}m` : `${minutes}m ${seconds}s`;
    }
    if (value >= 1000) {
      return `${Number((value / 1000).toFixed(1))}s`;
    }
    return `${value}ms`;
  }

  return new Intl.NumberFormat("en-US").format(value);
}

function renderSectionRecords(records: JsonObject): JSX.Element[] {
  return Object.entries(records).map(([key, value]) => (

      <strong>{formatLabel(key)}</strong>
      {renderValue(value)}

function renderValue(value: unknown): JSX.Element {
  if (Array.isArray(value) || (value && typeof value === "object")) {
    return (
      <pre className="inspector-record__code">
        <code>{JSON.stringify(value, null, 2)}</code>
      </pre>
    );
  }

  return <span className="inspector-record__text">{String(value)}</span>;

function renderRefList(title: string, refs: string[]): JSX.Element | null {
  if (refs.length === 0) {
    return null;

  return (
    <div className="inspector-record">
      <strong>{title}</strong>
      <ul className="inspector-ref-list">
        {refs.map((ref) => (
          <li key={ref}>{ref}</li>
        ))}
      </ul>
    </div>
  );
}

function hasSectionContent(section: InspectorSection): boolean {
  return (
    Object.keys(section.records).length > 0 ||
    section.stable_refs.length > 0 ||
    section.log_refs.length > 0 ||
    section.truncated ||
    section.redaction_status !== "none"
  );

    return {

    };
  }

    implementationPlan: detail.implementation_plan,
    toolConfirmationTraceRefs: detail.tool_confirmation_trace_refs,
    providerRetryTraceRefs: detail.provider_retry_trace_refs,
    providerCircuitBreakerTraceRefs: detail.provider_circuit_breaker_trace_refs,
    approvalResultRefs: detail.approval_result_refs,
  };

              {task.title}
            </li>

function formatLabel(value: string): string {
