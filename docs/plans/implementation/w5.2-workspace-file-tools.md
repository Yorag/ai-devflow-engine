## Key Behavior Decisions

- Production file tools are concrete `ToolProtocol` implementations only; they do not bypass `ToolRegistry.execute()`.
- Each tool constructor receives:
  - a `RunWorkspace`
  - a `WorkspaceManager`
  - an optional `RedactionPolicy`
- Each tool uses `WorkspaceManager.assert_inside_workspace()` on every declared path before touching the filesystem, even if the execution gate already ran the boundary port. This preserves safety when tools are called directly in tests and matches the split-plan requirement that tools only access the current run workspace.
- `read_file` decodes UTF-8 text only. If bytes cannot decode as UTF-8 or contain NUL bytes, the tool returns `tool_input_schema_invalid` only for schema problems and `tool_workspace_boundary_violation` only for boundary problems; binary-content rejection and missing-file failures must use existing catalog entries if available, otherwise fall back to `internal_error` with stable safe details. During implementation, if the current catalog lacks a suitable file-content/domain code and adding one would expand W5.2 beyond the split-plan’s file list, stop and report the mismatch.
- `write_file` creates missing parent directories under the workspace root and fully overwrites the target file content.
- `edit_file` requires:
  - `old_text` exists
  - `old_text` appears exactly once
  - replacing it once yields the expected output
  - otherwise the tool returns structured failure and does not mutate the file
- `glob` matches against workspace-relative POSIX-style paths, excludes directories by default from results, excludes any `RunWorkspace.excluded_relative_paths` entries such as `.runtime/logs`, and returns deterministic lexicographic ordering.
- `write_file` and `edit_file` successful results must include one stable side-effect ref each, shaped as `file-edit-trace:<tool_name>:<call_id>:<digest-prefix>`.
- Large text payloads and changed content must not be copied verbatim into previews longer than the redaction policy allows.

from __future__ import annotations

from dataclasses import dataclass, field
from fnmatch import fnmatch
from hashlib import sha256
from pathlib import Path
from typing import Any, Mapping

from backend.app.api.error_codes import ErrorCode
from backend.app.observability.redaction import RedactionPolicy
from backend.app.tools.protocol import (
    ToolBindableDescription,
    ToolError,
    ToolInput,
    ToolPermissionBoundary,
    ToolResult,
    ToolResultStatus,
    ToolRiskLevel,
    ToolSideEffectLevel,

from backend.app.workspace.manager import RunWorkspace, WorkspaceManager


def read_file(*, root: Path, relative_path: str) -> tuple[str, str]:
    path = root / relative_path
    raw = path.read_bytes()
    if b"\x00" in raw:
        raise ValueError("binary_content_blocked")
    try:
        content = raw.decode("utf-8")
    except UnicodeDecodeError as exc:
        raise ValueError("binary_content_blocked") from exc
    return relative_path, content


def _is_excluded_relative_path(
    relative_path: str,
    excluded_relative_paths: tuple[str, ...],
) -> bool:
    return any(
        relative_path == excluded
        or relative_path.startswith(f"{excluded}/")
        for excluded in excluded_relative_paths
    )


def glob(
    *,
    root: Path,
    pattern: str,
    excluded_relative_paths: tuple[str, ...],
) -> list[dict[str, str]]:
    matches: list[dict[str, str]] = []
    for candidate in sorted(root.rglob("*")):
        if not candidate.is_file():
            continue
        relative_path = candidate.relative_to(root).as_posix()
        if _is_excluded_relative_path(relative_path, excluded_relative_paths):
            continue
        if fnmatch(relative_path, pattern):
            matches.append({"path": relative_path, "path_type": "file"})
    return matches


@dataclass
class WorkspaceFileTool:
    workspace: RunWorkspace
    workspace_manager: WorkspaceManager
    redaction_policy: RedactionPolicy = RedactionPolicy()

    def _error_result(
        self,
        *,
        tool_input: ToolInput,
        tool_name: str,
        reason: str,
    ) -> ToolResult:
        return ToolResult(
            tool_name=tool_name,
            call_id=tool_input.call_id,
            status=ToolResultStatus.FAILED,
            error=ToolError.from_code(
                ErrorCode.INTERNAL_ERROR,
                trace_context=tool_input.trace_context,
                safe_details={"reason": reason},
            ),
            trace_context=tool_input.trace_context,
            coordination_key=tool_input.coordination_key,
        )

    def _preview(self, text: str) -> str:
        redacted = self.redaction_policy.summarize_text(
            text,
            payload_type="workspace_file_text",
        )
        return redacted.excerpt


@dataclass
class FileReadTool(WorkspaceFileTool):
    name: str = "read_file"
    category: str = "workspace"
    description: str = "Read one UTF-8 text file from the current run workspace."
    input_schema: dict[str, object] = field(default_factory=dict)
    result_schema: dict[str, object] = field(default_factory=dict)
    default_risk_level: ToolRiskLevel = ToolRiskLevel.READ_ONLY
    risk_categories: tuple = ()
    permission_boundary: ToolPermissionBoundary = ToolPermissionBoundary(
        boundary_type="workspace",
        requires_workspace=True,
        resource_scopes=("current_run_workspace",),
        workspace_target_paths=("path",),
    )
    side_effect_level: ToolSideEffectLevel = ToolSideEffectLevel.WORKSPACE_READ
    audit_required: bool = False
    schema_version: str = "tool-schema-v1"
    default_timeout_seconds: float | None = 5.0

    def __post_init__(self) -> None:
        self.input_schema = {
            "type": "object",
            "properties": {"path": {"type": "string", "minLength": 1}},
            "required": ["path"],
            "additionalProperties": False,
        }
        self.result_schema = {
            "type": "object",
            "properties": {
                "path": {"type": "string"},
                "content": {"type": "string"},
            },
            "required": ["path", "content"],
            "additionalProperties": False,
        }

    def bindable_description(self) -> ToolBindableDescription:
        return ToolBindableDescription(
            name=self.name,
            description=self.description,
            input_schema=dict(self.input_schema),
            result_schema=dict(self.result_schema),
            risk_level=self.default_risk_level,
            risk_categories=list(self.risk_categories),
            schema_version=self.schema_version,
            default_timeout_seconds=self.default_timeout_seconds,
        )

    def execute(self, tool_input: ToolInput) -> ToolResult:
        relative_path = str(tool_input.input_payload["path"])
        try:
            resolved = self.workspace_manager.assert_inside_workspace(
                relative_path,
                workspace=self.workspace,
                trace_context=tool_input.trace_context,
            )
            _, content = read_file(
                root=self.workspace.root,
                relative_path=resolved.relative_to(self.workspace.root).as_posix(),
            )
        except FileNotFoundError:
            return self._error_result(
                tool_input=tool_input,
                tool_name=self.name,
                reason="file_not_found",
            )
        except ValueError as exc:
            return self._error_result(
                tool_input=tool_input,
                tool_name=self.name,
                reason=str(exc),
            )
        return ToolResult(
            tool_name=self.name,
            call_id=tool_input.call_id,
            status=ToolResultStatus.SUCCEEDED,
            output_payload={"path": relative_path, "content": content},
            output_preview=self._preview(content),
            trace_context=tool_input.trace_context,
            coordination_key=tool_input.coordination_key,
        )

def write_file(*, root: Path, relative_path: str, content: str) -> None:
    path = root / relative_path
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(content, encoding="utf-8")


def edit_file(*, root: Path, relative_path: str, old_text: str, new_text: str) -> str:
    path = root / relative_path
    content = path.read_text(encoding="utf-8")
    occurrences = content.count(old_text)
    if occurrences == 0:
        raise ValueError("edit_target_not_found")
    if occurrences != 1:
        raise ValueError("edit_target_not_unique")
    updated = content.replace(old_text, new_text, 1)
    path.write_text(updated, encoding="utf-8")
    return updated


def _file_edit_ref(*, tool_name: str, call_id: str, path: str, content: str) -> str:
    digest = sha256(f"{path}\\n{content}".encode("utf-8")).hexdigest()[:12]
    return f"file-edit-trace:{tool_name}:{call_id}:{digest}"

def glob(
    *,
    root: Path,
    pattern: str,
    excluded_relative_paths: tuple[str, ...],
) -> list[dict[str, str]]:
    matches: list[dict[str, str]] = []
    for candidate in sorted(root.rglob("*")):
        if not candidate.is_file():
            continue
        relative_path = candidate.relative_to(root).as_posix()
        if _is_excluded_relative_path(relative_path, excluded_relative_paths):
            continue
        if fnmatch(relative_path, pattern):
            matches.append({"path": relative_path, "path_type": "file"})
    return matches
