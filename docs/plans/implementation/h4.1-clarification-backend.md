**Tech Stack:** Python 3.11+, FastAPI, SQLAlchemy, Pydantic v2, existing multi-SQLite `DatabaseManager`, existing `RuntimeOrchestrationService`, existing `AuditService`, `pytest` through `uv run`.

## Log & Audit Integration

H4.1 is a user-command and run-lifecycle slice, so audit records and trace metadata are required.

- Runtime log category:
  - Use existing `AuditService` JSONL copy path. Do not create a second log writer or a separate clarification log table.
- Audit actions:
  - `clarification.request`
  - `session.message.clarification_reply`
  - `session.message.clarification_reply.rejected`
  - `session.message.clarification_reply.resume_failed`
- Associated objects:
  - target type `clarification` for request and resume failure.
  - target type `session` for user reply success and illegal-state rejection.
- Result semantics:
  - Request and reply success use `AuditResult.SUCCEEDED`.
  - Illegal state uses `AuditResult.REJECTED`.
  - Runtime resume failure uses `AuditResult.FAILED`.
- Trace fields:
  - Use the caller `TraceContext` and attach `session_id`, `run_id`, and `stage_run_id` with `child_span()`.
  - Preserve `request_id`, `trace_id`, `correlation_id`, and span ancestry.
  - Runtime boundary child spans remain owned by `RuntimeOrchestrationService`.
- Payload safety:
  - Audit metadata includes stable ids, status values, stage type, `clarification_id`, `control_record_id`, `graph_interrupt_ref`, and answer length.
  - Audit metadata must not store raw secrets or credentials.
  - User answer may appear as a business fact in `ClarificationRecord.answer` and domain event `message_item.content`; audit metadata stores only `answer_length`.
- Failure behavior:
  - Required audit failures for user-command success or rejection must reject or roll back the operation through `AuditService.require_audit_record()` / `record_rejected_command()`.
  - Runtime `resume_interrupt()` failure must be audited with `record_failed_command()` and roll back answer/status/event changes before raising a service error.
  - Log or audit records never replace `ClarificationRecordModel`, `RunControlRecordModel`, domain events, or status projections as product facts.

from collections.abc import Callable
from dataclasses import dataclass

from sqlalchemy.orm import Session

from backend.app.db.models.control import SessionModel

@dataclass(frozen=True)
class RunLifecycleProjection:
    run: PipelineRunModel
    stage: StageRunModel
    session: SessionModel


class RunLifecycleError(RuntimeError):
    pass


class RunLifecycleService:
    def __init__(
        self,
        *,
        runtime_session: Session,
        control_session: Session,
        now: Callable[[], datetime] | None = None,
    ) -> None:
        self._runtime_session = runtime_session
        self._control_session = control_session
        self._now = now or (lambda: datetime.now(UTC))

    def mark_waiting_clarification(
        self,
        *,
        session_id: str,
        run_id: str,
        stage_run_id: str,
    ) -> RunLifecycleProjection:
        projection = self._load_projection(
            session_id=session_id,
            run_id=run_id,
            stage_run_id=stage_run_id,
        )
        if projection.stage.stage_type is not StageType.REQUIREMENT_ANALYSIS:
            raise RunLifecycleError("Clarification is only valid in requirement_analysis.")
        if projection.stage.status is not StageStatus.RUNNING:
            raise RunLifecycleError("Clarification can be requested only from a running stage.")
        projection.run.status = RunStateMachine.transition(
            projection.run.status,
            RunTransitionEvent.REQUEST_CLARIFICATION,
        )
        projection.stage.status = StageStatus.WAITING_CLARIFICATION
        projection.session.status = RunStateMachine.project_session_status(projection.run.status)
        projection.session.latest_stage_type = StageType.REQUIREMENT_ANALYSIS
        self._touch(projection)
        return projection

    def mark_running_after_clarification_reply(
        self,
        *,
        session_id: str,
        run_id: str,
        stage_run_id: str,
    ) -> RunLifecycleProjection:
        projection = self._load_projection(
            session_id=session_id,
            run_id=run_id,
            stage_run_id=stage_run_id,
        )
        if projection.session.status is not SessionStatus.WAITING_CLARIFICATION:
            raise RunLifecycleError("clarification_reply requires Session.status = waiting_clarification.")
        if projection.run.status is not RunStatus.WAITING_CLARIFICATION:
            raise RunLifecycleError("clarification_reply requires PipelineRun.status = waiting_clarification.")
        if projection.stage.stage_type is not StageType.REQUIREMENT_ANALYSIS:
            raise RunLifecycleError("clarification_reply is only valid in requirement_analysis.")
        if projection.stage.status is not StageStatus.WAITING_CLARIFICATION:
            raise RunLifecycleError("clarification_reply requires waiting clarification stage.")
        projection.run.status = RunStateMachine.transition(
            projection.run.status,
            RunTransitionEvent.CLARIFICATION_REPLY,
        )
        projection.stage.status = StageStatus.RUNNING
        projection.session.status = RunStateMachine.project_session_status(projection.run.status)
        projection.session.latest_stage_type = StageType.REQUIREMENT_ANALYSIS
        self._touch(projection)
        return projection

    def _load_projection(
        self,
        *,
        session_id: str,
        run_id: str,
        stage_run_id: str,
    ) -> RunLifecycleProjection:
        session_model = self._control_session.get(SessionModel, session_id)
        run = self._runtime_session.get(PipelineRunModel, run_id)
        stage = self._runtime_session.get(StageRunModel, stage_run_id)
        if session_model is None or run is None or stage is None:
            raise RunLifecycleError("Run lifecycle objects were not found.")
        if session_model.current_run_id != run.run_id:
            raise RunLifecycleError("Session current_run_id does not match run.")
        if run.session_id != session_model.session_id:
            raise RunLifecycleError("Run session_id does not match session.")
        if run.current_stage_run_id != stage.stage_run_id:
            raise RunLifecycleError("Run current_stage_run_id does not match stage.")
        if stage.run_id != run.run_id:
            raise RunLifecycleError("Stage run_id does not match run.")
        return RunLifecycleProjection(run=run, stage=stage, session=session_model)

    def _touch(self, projection: RunLifecycleProjection) -> None:
        timestamp = self._now()
        projection.run.updated_at = timestamp
        projection.stage.updated_at = timestamp
        projection.session.updated_at = timestamp
        self._runtime_session.add(projection.run)
        self._runtime_session.add(projection.stage)
        self._control_session.add(projection.session)

    "RunLifecycleService",

def build_app(tmp_path: Path):
    default_root = tmp_path / "ai-devflow-engine"
    default_root.mkdir()
    app = create_app(
        EnvironmentSettings(
            platform_runtime_root=tmp_path / "runtime",
            default_project_root=default_root,
        )
    )
    ControlBase.metadata.create_all(app.state.database_manager.engine(DatabaseRole.CONTROL))
    RuntimeBase.metadata.create_all(app.state.database_manager.engine(DatabaseRole.RUNTIME))
    EventBase.metadata.create_all(app.state.database_manager.engine(DatabaseRole.EVENT))
    LogBase.metadata.create_all(app.state.database_manager.engine(DatabaseRole.LOG))
    app.state.h41_runtime_port = FakeRuntimePort()
    app.state.h41_checkpoint_port = FakeCheckpointPort()
    return app

def seed_waiting_clarification_via_service(app) -> str:
    from backend.tests.services.test_clarification_flow import build_trace
    from backend.tests.services.test_clarification_flow import seed_waiting_requirement_analysis
    from backend.app.services.clarifications import ClarificationService
    from backend.app.services.runtime_orchestration import RuntimeOrchestrationService

    manager = app.state.database_manager
    seed_waiting_requirement_analysis(manager)
    runtime_port = app.state.h41_runtime_port
    service = ClarificationService(
        control_session=manager.session(DatabaseRole.CONTROL),
        runtime_session=manager.session(DatabaseRole.RUNTIME),
        event_session=manager.session(DatabaseRole.EVENT),
        audit_service=RecordingAuditService(),
        runtime_orchestration=RuntimeOrchestrationService(
            runtime_port=runtime_port,
            checkpoint_port=app.state.h41_checkpoint_port,
        ),
    )
    result = service.request_clarification(
        session_id="session-1",
        run_id="run-1",
        stage_run_id="stage-run-1",
        question="Which package should be changed?",
        payload_ref="clarification-payload-1",
        trace_context=build_trace(),
    )
    return result.clarification_id


def test_post_session_message_accepts_clarification_reply_and_restores_running(tmp_path: Path) -> None:
    app = build_app(tmp_path)
    clarification_id = seed_waiting_clarification_via_service(app)

    with TestClient(app) as client:
        response = client.post(
            "/api/sessions/session-1/messages",
            json={"message_type": "clarification_reply", "content": "Change backend only."},
            headers={
                "X-Request-ID": "req-clarification-reply",
                "X-Correlation-ID": "corr-clarification-reply",
            },
        )

    assert response.status_code == 200
    body = response.json()
    assert body["session"]["status"] == "running"
    assert body["message_item"]["type"] == "user_message"
    assert body["message_item"]["content"] == "Change backend only."
    assert body["message_item"]["stage_run_id"] == "stage-run-1"
    with app.state.database_manager.session(DatabaseRole.RUNTIME) as session:
        clarification = session.get(ClarificationRecordModel, clarification_id)
        run = session.get(PipelineRunModel, "run-1")
        assert clarification.answer == "Change backend only."
        assert run.status is RunStatus.RUNNING
    with app.state.database_manager.session(DatabaseRole.EVENT) as session:
        event = (
            session.query(DomainEventModel)
            .filter(DomainEventModel.event_type == SseEventType.CLARIFICATION_ANSWERED)
            .one()
        )
        assert event.payload["message_item"]["content"] == "Change backend only."
    with app.state.database_manager.session(DatabaseRole.LOG) as session:
        audit = (
            session.query(AuditLogEntryModel)
            .filter(AuditLogEntryModel.action == "session.message.clarification_reply")
            .one()
        )
        assert audit.request_id == "req-clarification-reply"
        assert audit.correlation_id == "corr-clarification-reply"


def test_clarification_reply_rejects_illegal_state_with_conflict_and_audit(tmp_path: Path) -> None:
    app = build_app(tmp_path)

    with TestClient(app) as client:
        create_response = client.post("/api/projects/project-default/sessions")
        created = create_response.json()
        response = client.post(
            f"/api/sessions/{created['session_id']}/messages",
            json={"message_type": "clarification_reply", "content": "Too early."},
            headers={
                "X-Request-ID": "req-clarification-conflict",
                "X-Correlation-ID": "corr-clarification-conflict",
            },
        )

    assert response.status_code == 409
    assert response.json()["error_code"] == "validation_error"
    assert "waiting_clarification" in response.json()["message"]
    with app.state.database_manager.session(DatabaseRole.LOG) as session:
        audit = (
            session.query(AuditLogEntryModel)
            .filter(AuditLogEntryModel.action == "session.message.clarification_reply.rejected")
            .one()
        )
        assert audit.request_id == "req-clarification-conflict"
        assert audit.correlation_id == "corr-clarification-conflict"
