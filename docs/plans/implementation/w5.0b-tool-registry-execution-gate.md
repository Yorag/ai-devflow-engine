**Tech Stack:** Python 3.11+, Pydantic v2, dataclasses, pytest, existing `TraceContext`, `ErrorCode`, `ToolProtocol`, `ToolResult`, and backend enum contracts.

## Log & Audit Integration

- Runtime log category: `tool`-scope runtime summary through an injected run-log port. This slice defines the port and calls it for accepted execution results and refusals; concrete JSONL/index persistence remains existing observability infrastructure and later runtime integration.
- Audit action:
  - `tool.intent` for side-effect or audit-required tools before execution.
  - `tool.rejected` for workspace-boundary or audit-sensitive refusal when an audit recorder is available.
- Correlation fields: every result and error keeps `request_id`, `trace_id`, `correlation_id`, `span_id`, `parent_span_id`, `run_id`, and `stage_run_id` through `TraceContext`.
- Sensitive data: errors use W5.0a `ToolError.from_code()` and safe `safe_details`; tests use path and field names only.
- Audit failure behavior: when a tool requires audit or side-effect intent and no audit recorder can produce a `ToolAuditRef`, the gate returns `tool_audit_required_failed` and does not execute the concrete tool.
- Product truth boundary: runtime logs and audit records do not replace `ToolResult`, stage artifacts, narrative feed entries, inspector projections, or future `ToolConfirmationRequest`.

class RecordingLog:
    def __init__(self) -> None:

class RecordingAudit:
    def __init__(self) -> None:
        self.intents: list[str] = []
        self.rejections: list[str] = []

    def record_tool_intent(
        self,
        *,
        request: ToolExecutionRequest,
        tool_name: str,
        trace_context: TraceContext,
    ) -> ToolAuditRef:
        self.intents.append(tool_name)
        return ToolAuditRef(
            audit_id=f"audit-{request.call_id}",
            action="tool.intent",
            trace_id=trace_context.trace_id,
            correlation_id=trace_context.correlation_id,
            metadata_ref=f"payload-{request.call_id}",
        )

        self,
        *,
        request: ToolExecutionRequest,
        error_code: ErrorCode,
        trace_context: TraceContext,
    ) -> ToolAuditRef:
        self.rejections.append(error_code.value)
        return ToolAuditRef(
            audit_id=f"audit-reject-{request.call_id}",
            action="tool.rejected",
            trace_id=trace_context.trace_id,
            correlation_id=trace_context.correlation_id,
            metadata_ref=f"payload-reject-{request.call_id}",
        )


class WorkspaceBoundary:
    def __init__(self, blocked_target: str | None = None) -> None:
        self.checked_targets: list[str] = []
        self.blocked_target = blocked_target

    def assert_inside_workspace(
        self,
        target: str,
        *,
        trace_context: TraceContext,
    ) -> None:
        self.checked_targets.append(target)
        if target == self.blocked_target:
            raise ToolWorkspaceBoundaryError(
                "Tool target is outside the run workspace.",
                target=target,
            )


class RecordingRiskHook:

        self,

        request: ToolExecutionRequest,
        tool_name: str,
        trace_context: TraceContext,
    ) -> None:
        self.calls.append(f"{tool_name}:{request.call_id}:{trace_context.span_id}")

from __future__ import annotations

import re
import time
from dataclasses import dataclass, field
from typing import Any, Mapping, Protocol

from pydantic import BaseModel, ConfigDict, Field

from backend.app.api.error_codes import ErrorCode
from backend.app.domain.enums import StageType
from backend.app.domain.trace_context import TraceContext
from backend.app.tools.protocol import (
    ToolAuditRef,
    ToolError,
    ToolInput,
    ToolProtocol,
    ToolResult,
    ToolResultStatus,
    ToolSideEffectLevel,
)


JsonObject = dict[str, Any]
_CONTRACT_NAME_PATTERN = re.compile(r"^[a-z][a-z0-9_]*$")
_PATH_FIELD_NAMES = {"path", "target_path", "file_path", "working_dir", "cwd"}


class ToolWorkspaceBoundaryError(ValueError):
    def __init__(self, message: str, *, target: str) -> None:
        super().__init__(message)
        self.target = target


class WorkspaceBoundaryPort(Protocol):
    def assert_inside_workspace(
        self,
        target: str,
        *,
        trace_context: TraceContext,
    ) -> None: ...


class ToolAuditRecorderPort(Protocol):
    def record_tool_intent(
        self,
        *,
        request: "ToolExecutionRequest",
        tool_name: str,
        trace_context: TraceContext,
    ) -> ToolAuditRef: ...

    def record_tool_rejection(
        self,
        *,
        request: "ToolExecutionRequest",
        error_code: ErrorCode,
        trace_context: TraceContext,
    ) -> ToolAuditRef: ...


class ToolRunLogRecorderPort(Protocol):
    def record_tool_result(
        self,
        *,
        request: "ToolExecutionRequest",
        result: ToolResult,
        duration_ms: int,
    ) -> None: ...


class ToolRiskInspectionPort(Protocol):
    def inspect_tool_intent(
        self,
        *,
        request: "ToolExecutionRequest",
        tool_name: str,
        trace_context: TraceContext,
    ) -> None: ...


class _StrictBaseModel(BaseModel):
    model_config = ConfigDict(extra="forbid")


class ToolExecutionRequest(_StrictBaseModel):
    tool_name: str = Field(min_length=1)
    call_id: str = Field(min_length=1)
    input_payload: JsonObject = Field(default_factory=dict)
    trace_context: TraceContext
    coordination_key: str = Field(min_length=1)
    timeout_seconds: float | None = Field(default=None, gt=0)


@dataclass(frozen=True)
class ToolExecutionContext:
    stage_type: StageType
    stage_contracts: Mapping[str, Any]
    trace_context: TraceContext
    runtime_tool_timeout_seconds: float | None = None
    platform_tool_timeout_hard_limit_seconds: float | None = None
    workspace_boundary: WorkspaceBoundaryPort | None = None
    audit_recorder: ToolAuditRecorderPort | None = None
    run_log_recorder: ToolRunLogRecorderPort | None = None
    risk_policy: ToolRiskInspectionPort | None = None

    @property
    def allowed_tools(self) -> tuple[str, ...]:
        contract = self.stage_contracts.get(self.stage_type.value, {})
        raw_allowed = contract.get("allowed_tools", [])
        if not isinstance(raw_allowed, list):
            return ()
        return tuple(item for item in raw_allowed if isinstance(item, str))
