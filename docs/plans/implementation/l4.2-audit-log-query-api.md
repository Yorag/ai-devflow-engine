## Log And Audit Integration

This slice is read-only. It must preserve the existing audit boundary rather than reinterpret audit rows as product truth.

- `AuditLogEntryModel` remains a security/operations ledger, not a replacement for approvals, tool confirmations, delivery records, runs, or feed entries.
- The query response may expose `reason`, `metadata_excerpt`, `metadata_ref`, `request_id`, `correlation_id`, `trace_id`, `span_id`, and object identifiers because these are already part of the approved audit projection contract.
- The route must not expose raw `LogPayloadModel.summary`, full metadata bodies, JSONL file contents, replay commands, delete commands, or mutation endpoints.
- If canonical runtime settings cannot be read or validated, the query must fail with `config_snapshot_unavailable` instead of silently falling back to a hard-coded `limit`.

```python
def list_audit_logs(
    self,
    *,
    actor_type: AuditActorType | None = None,
    action: str | None = None,
    target_type: str | None = None,
    target_id: str | None = None,
    run_id: str | None = None,
    stage_run_id: str | None = None,
    correlation_id: str | None = None,
    result: AuditResult | None = None,
    since: datetime | None = None,
    until: datetime | None = None,
    cursor: str | None = None,
    limit: int | None = None,
) -> AuditLogQueryResponse:
    effective_limit = self._effective_limit(limit)
    decoded_cursor = self._decode_cursor(cursor) if cursor is not None else None
    query_echo = self._query_echo(
        actor_type=actor_type,
        action=action,
        target_type=target_type,
        target_id=target_id,
        run_id=run_id,
        stage_run_id=stage_run_id,
        correlation_id=correlation_id,
        result=result,
        since=since,
        until=until,
        cursor=cursor,
        limit=effective_limit,
    )
    statement = select(AuditLogEntryModel)
    if actor_type is not None:
        statement = statement.where(AuditLogEntryModel.actor_type == actor_type)
    if action is not None:
        statement = statement.where(AuditLogEntryModel.action == action)
    if target_type is not None:
        statement = statement.where(AuditLogEntryModel.target_type == target_type)
    if target_id is not None:
        statement = statement.where(AuditLogEntryModel.target_id == target_id)
    if run_id is not None:
        statement = statement.where(AuditLogEntryModel.run_id == run_id)
    if stage_run_id is not None:
        statement = statement.where(AuditLogEntryModel.stage_run_id == stage_run_id)
    if correlation_id is not None:
        statement = statement.where(AuditLogEntryModel.correlation_id == correlation_id)
    if result is not None:
        statement = statement.where(AuditLogEntryModel.result == result)
    if since is not None:
        statement = statement.where(AuditLogEntryModel.created_at >= since)
    if until is not None:
        statement = statement.where(AuditLogEntryModel.created_at <= until)
    if decoded_cursor is not None:
        cursor_created_at, cursor_audit_id = decoded_cursor
        statement = statement.where(
            or_(
                AuditLogEntryModel.created_at < cursor_created_at,
                and_(
                    AuditLogEntryModel.created_at == cursor_created_at,
                    AuditLogEntryModel.audit_id < cursor_audit_id,
                ),
            )
        )
    statement = statement.order_by(
        AuditLogEntryModel.created_at.desc(),
        AuditLogEntryModel.audit_id.desc(),
    ).limit(effective_limit + 1)
    rows = list(self._session.scalars(statement))
    page_rows = rows[:effective_limit]
    next_cursor = (
        self._encode_cursor(page_rows[-1].created_at, page_rows[-1].audit_id)
        if len(rows) > effective_limit and page_rows
        else None
    )
    return AuditLogQueryResponse(
        entries=[self._to_audit_projection(row) for row in page_rows],
        next_cursor=next_cursor,
        has_more=len(rows) > effective_limit,
        query=query_echo,
    )
```
