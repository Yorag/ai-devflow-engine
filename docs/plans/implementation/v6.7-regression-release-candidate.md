def _feed_identities(entries: list[dict[str, Any]]) -> list[tuple[str, str]]:
    identities: list[tuple[str, str]] = []
    for entry in entries:
        entry_type = entry["type"]
        if entry_type == "delivery_result":
            identities.append((entry_type, entry["delivery_record_id"]))
        elif entry_type == "approval_request":
            identities.append((entry_type, entry["approval_id"]))
        elif entry_type == "approval_result":
            identities.append((entry_type, entry["approval_id"]))
        elif entry_type == "tool_confirmation":
            identities.append((entry_type, entry["tool_confirmation_id"]))
        elif entry_type == "stage_node":
            identities.append((entry_type, entry["stage_run_id"]))
        else:
            identities.append((entry_type, entry["entry_id"]))
    return identities
