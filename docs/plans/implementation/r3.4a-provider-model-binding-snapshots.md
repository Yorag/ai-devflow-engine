**Goal:** Freeze Provider and model binding runtime semantics before run startup by creating structured `ProviderSnapshot` and `ModelBindingSnapshot` rows associated with a `PipelineRun`.

**Architecture:** Snapshot builders live in `backend/app/domain/provider_snapshot.py` and convert already-selected control-plane Provider rows plus template/internal binding selections into immutable per-run snapshot objects. `RuntimeSnapshotRepository` persists those objects to `runtime.db`, and `RunLifecycleService.attach_provider_snapshots()` validates and saves the snapshot rows without adding new `PipelineRunModel` columns. This slice does not implement first-run startup, Provider API changes, runtime-limit snapshots, graph compilation, LangGraph runtime, or Provider adapters.

## Design Decisions

- `PipelineRunModel` already associates Provider/model binding snapshots through `ProviderSnapshotModel.run_id` and `ModelBindingSnapshotModel.run_id`. R3.4a therefore uses these runtime snapshot rows as the stable association and does not add Provider/model binding reference columns to `PipelineRunModel`.
- `ProviderSnapshotBuilder` only snapshots Provider rows explicitly required by the frozen template and internal binding selections. It does not snapshot every Provider in the control database.
- `ModelBindingSnapshotBuilder` derives `agent_role` bindings from `TemplateSnapshot.stage_role_bindings`. It requires explicit `InternalModelBindingSelection` entries for `context_compression`, `structured_output_repair`, and `validation_pass`; the slice does not invent a new editable platform configuration source for those bindings.
- Provider `source_config_version` uses the source `ProviderModel.updated_at` timestamp in UTC ISO form because the current Provider control model has no separate version field.
- Credential values are never read or stored. The snapshot stores only `api_key_ref` and rejects credential refs that do not match the allowed `env:` prefixes.

def _snapshot_id(prefix: str, run_id: str, suffix: str) -> str:
    raw = f"{prefix}-{run_id}-{suffix}"
    if len(raw) <= 80:
        return raw
    digest = sha256(raw.encode("utf-8")).hexdigest()[:16]
    return f"{prefix}-{digest}"


def _utc_source_version(value: datetime) -> str:
    if value.tzinfo is None:
        value = value.replace(tzinfo=UTC)
    return value.isoformat()


def _is_safe_api_key_ref(value: str | None, prefixes: tuple[str, ...]) -> bool:
    if value is None:
        return True
    env_name = value.removeprefix("env:")
    env_name_has_valid_chars = all(
        char == "_" or char.isascii() and char.isalnum() for char in env_name
    )
    return (
        value.startswith("env:")
        and bool(env_name)
        and env_name_has_valid_chars
        and any(env_name.startswith(prefix) for prefix in prefixes)
    )


class ProviderSnapshotBuilder:
    @classmethod
    def build_for_run(
        cls,
        providers: Iterable[Any],
        *,
        run_id: str,
        required_provider_ids: Iterable[str],
        created_at: datetime,
        credential_env_prefixes: Iterable[str] | None = None,
    ) -> tuple[ProviderSnapshot, ...]:
        prefixes = tuple(
            credential_env_prefixes
            if credential_env_prefixes is not None
            else EnvironmentSettings().credential_env_prefixes
        )
        providers_by_id = {provider.provider_id: provider for provider in providers}
        missing = sorted(set(required_provider_ids) - set(providers_by_id))
        if missing:
            raise ProviderSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                f"ProviderSnapshot cannot be built because providers are missing: {', '.join(missing)}.",
            )

        snapshots = []
        for provider_id in sorted(set(required_provider_ids)):
            provider = providers_by_id[provider_id]
            snapshots.append(
                cls._build_one(
                    provider,
                    run_id=run_id,
                    created_at=created_at,
                    credential_env_prefixes=prefixes,
                )
            )
        return tuple(snapshots)

    @classmethod
    def _build_one(
        cls,
        provider: Any,
        *,
        run_id: str,
        created_at: datetime,
        credential_env_prefixes: tuple[str, ...],
    ) -> ProviderSnapshot:
        api_key_ref = getattr(provider, "api_key_ref")
        if not _is_safe_api_key_ref(api_key_ref, credential_env_prefixes):
            raise ProviderSnapshotBuilderError(
                ErrorCode.CONFIG_CREDENTIAL_ENV_NOT_ALLOWED,
                "ProviderSnapshot cannot be built because the credential reference is not allowed.",
            )
        model_id = cls._require_string(provider, "default_model_id")
        capabilities = cls._capabilities_for_model(provider, model_id)
        return ProviderSnapshot(
            snapshot_id=_snapshot_id("provider-snapshot", run_id, provider.provider_id),
            run_id=run_id,
            provider_id=cls._require_string(provider, "provider_id"),
            display_name=cls._require_string(provider, "display_name"),
            provider_source=ProviderSource(provider.provider_source),
            protocol_type=ProviderProtocolType(provider.protocol_type),
            base_url=cls._require_string(provider, "base_url"),
            api_key_ref=api_key_ref,
            model_id=model_id,
            capabilities=capabilities,
            source_config_version=_utc_source_version(provider.updated_at),
            created_at=created_at,
        )

    @staticmethod
    def _require_string(source: Any, field_name: str) -> str:
        value = getattr(source, field_name)
        if not isinstance(value, str) or not value.strip():
            raise ProviderSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                f"ProviderSnapshot cannot be built because {field_name} is invalid.",
            )
        return value

    @staticmethod
    def _capabilities_for_model(
        provider: Any,
        model_id: str,
    ) -> SnapshotModelRuntimeCapabilities:
        capabilities = getattr(provider, "runtime_capabilities")
        if not isinstance(capabilities, list):
            raise ProviderSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                "ProviderSnapshot cannot be built because runtime capabilities are invalid.",
            )
        matches = [
            item
            for item in capabilities
            if isinstance(item, Mapping) and item.get("model_id") == model_id
        ]
        if len(matches) != 1:
            raise ProviderSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                f"ProviderSnapshot cannot be built because capabilities for model {model_id} are unavailable.",
            )
        try:
            return SnapshotModelRuntimeCapabilities.model_validate(matches[0])
        except Exception as exc:
            raise ProviderSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                f"ProviderSnapshot cannot be built because capabilities for model {model_id} are invalid.",
            ) from exc

class ModelBindingSnapshotBuilder:
    @classmethod
    def build_for_run(
        cls,
        template_snapshot: TemplateSnapshot,
        *,
        provider_snapshots: Iterable[ProviderSnapshot],
        internal_bindings: Iterable[InternalModelBindingSelection],
        run_id: str,
        created_at: datetime,
    ) -> tuple[ModelBindingSnapshot, ...]:
        if template_snapshot.run_id != run_id:
            raise ModelBindingSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                "ModelBindingSnapshot run_id must match TemplateSnapshot run_id.",
            )
        providers_by_id = {
            snapshot.provider_id: snapshot for snapshot in provider_snapshots
        }
        results: list[ModelBindingSnapshot] = []
        for index, binding in enumerate(template_snapshot.stage_role_bindings):
            provider = cls._provider_for(providers_by_id, binding.provider_id)
            results.append(
                ModelBindingSnapshot(
                    snapshot_id=_snapshot_id(
                        "model-binding-snapshot",
                        run_id,
                        f"agent-role-{index + 1}",
                    ),
                    run_id=run_id,
                    binding_id=f"agent-role-{binding.stage_type.value}",
                    binding_type="agent_role",
                    stage_type=binding.stage_type,
                    role_id=binding.role_id,
                    provider_snapshot_id=provider.snapshot_id,
                    provider_id=provider.provider_id,
                    model_id=provider.model_id,
                    capabilities=provider.capabilities,
                    model_parameters={},
                    source_config_version=template_snapshot.snapshot_ref,
                    created_at=created_at,
                )
            )
        internal = cls._required_internal_bindings(internal_bindings)
        for index, selection in enumerate(internal, start=1):
            provider = cls._provider_for(providers_by_id, selection.provider_id)
            model_id = selection.model_id or provider.model_id
            if model_id != provider.model_id:
                raise ModelBindingSnapshotBuilderError(
                    ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                    "Internal model binding must reference a frozen ProviderSnapshot model.",
                )
            results.append(
                ModelBindingSnapshot(
                    snapshot_id=_snapshot_id(
                        "model-binding-snapshot",
                        run_id,
                        f"internal-{index}",
                    ),
                    run_id=run_id,
                    binding_id=selection.binding_id,
                    binding_type=selection.binding_type,
                    stage_type=None,
                    role_id=None,
                    provider_snapshot_id=provider.snapshot_id,
                    provider_id=provider.provider_id,
                    model_id=provider.model_id,
                    capabilities=provider.capabilities,
                    model_parameters=dict(selection.model_parameters),
                    source_config_version="internal-model-binding-v1",
                    created_at=created_at,
                )
            )
        return tuple(results)

    @staticmethod
    def _provider_for(
        providers_by_id: dict[str, ProviderSnapshot],
        provider_id: str,
    ) -> ProviderSnapshot:
        provider = providers_by_id.get(provider_id)
        if provider is None:
            raise ModelBindingSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                f"ModelBindingSnapshot cannot be built because ProviderSnapshot {provider_id} is missing.",
            )
        return provider

    @staticmethod
    def _required_internal_bindings(
        selections: Iterable[InternalModelBindingSelection],
    ) -> tuple[InternalModelBindingSelection, ...]:
        by_type = {selection.binding_type: selection for selection in selections}
        missing = [
            binding_type
            for binding_type in INTERNAL_MODEL_BINDING_TYPES
            if binding_type not in by_type
        ]
        if missing:
            raise ModelBindingSnapshotBuilderError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                f"ModelBindingSnapshot internal bindings are missing: {', '.join(missing)}.",
            )
        return tuple(by_type[binding_type] for binding_type in INTERNAL_MODEL_BINDING_TYPES)

    def attach_provider_snapshots(
        self,
        run: PipelineRunModel,
        *,
        provider_snapshots: tuple[ProviderSnapshot, ...],
        model_binding_snapshots: tuple[ModelBindingSnapshot, ...],
    ) -> PipelineRunModel:
        if not provider_snapshots:
            raise ValueError("provider_snapshots must not be empty")
        if not model_binding_snapshots:
            raise ValueError("model_binding_snapshots must not be empty")
        if any(snapshot.run_id != run.run_id for snapshot in provider_snapshots):
            raise ValueError("ProviderSnapshot run_id must match PipelineRun.run_id")
        if any(snapshot.run_id != run.run_id for snapshot in model_binding_snapshots):
            raise ValueError(
                "ModelBindingSnapshot run_id must match PipelineRun.run_id"
            )
        provider_snapshot_ids = {
            snapshot.snapshot_id for snapshot in provider_snapshots
        }
        missing_provider_snapshot_ids = sorted(
            {
                snapshot.provider_snapshot_id
                for snapshot in model_binding_snapshots
                if snapshot.provider_snapshot_id not in provider_snapshot_ids
            }
        )
        if missing_provider_snapshot_ids:
            raise ValueError(
                "ModelBindingSnapshot provider_snapshot_id must reference the "
                f"same run snapshots: {', '.join(missing_provider_snapshot_ids)}"
            )

        repository = RuntimeSnapshotRepository(self._runtime_session)
        for snapshot in provider_snapshots:
            repository.save_provider_snapshot(snapshot)
        for snapshot in model_binding_snapshots:
            repository.save_model_binding_snapshot(snapshot)
        run.updated_at = self._now()
        self._runtime_session.add(run)
        return run
