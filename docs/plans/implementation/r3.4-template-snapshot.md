> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. Repository override: do not commit, branch, merge, rebase, push, update coordination store, or mark platform/split tasks complete.

**Goal:** Freeze the selected `PipelineTemplateModel` into an immutable per-run `TemplateSnapshot` before a run starts, then attach its stable reference to `PipelineRunModel`.

**Architecture:** `TemplateSnapshotBuilder` lives in `backend/app/domain/template_snapshot.py` and converts an existing control-plane template row into immutable, validated snapshot data. `RunLifecycleService.attach_template_snapshot()` lives in `backend/app/services/runs.py` and is the narrow run lifecycle attach point that writes the snapshot reference to an existing `PipelineRunModel`; it does not implement first-run API startup, provider/model snapshots, runtime-limit snapshots, graph compilation, or persistence migrations.

**Tech Stack:** Python 3.11+, Pydantic v2 models, SQLAlchemy ORM models, pytest via `uv run python -m pytest`.

**Execution Status:** Local worker result `reported`; expected committed ingest result `implemented`.

def build_trace() -> TraceContext:

def build_manager(tmp_path: Path) -> DatabaseManager:
    manager = DatabaseManager(
        _database_paths={role: tmp_path / f"{role.value}.db" for role in DatabaseRole},

    )
    ControlBase.metadata.create_all(manager.engine(DatabaseRole.CONTROL))
    RuntimeBase.metadata.create_all(manager.engine(DatabaseRole.RUNTIME))

from __future__ import annotations

from datetime import datetime
from typing import Any

from pydantic import BaseModel, ConfigDict, Field, field_validator

from backend.app.domain.enums import ApprovalType, StageType, TemplateSource
from backend.app.schemas.template import (
    FIXED_APPROVAL_CHECKPOINTS,
    FIXED_STAGE_SEQUENCE,
)


TEMPLATE_SNAPSHOT_SCHEMA_VERSION = "template-snapshot-v1"


class StageRoleSnapshot(BaseModel):
    model_config = ConfigDict(extra="forbid", frozen=True)

    stage_type: StageType
    role_id: str = Field(min_length=1)
    system_prompt: str = Field(min_length=1)
    provider_id: str = Field(min_length=1)


class TemplateSnapshot(BaseModel):
    model_config = ConfigDict(extra="forbid", frozen=True)

    snapshot_ref: str = Field(min_length=1)
    run_id: str = Field(min_length=1)
    source_template_id: str = Field(min_length=1)
    source_template_name: str = Field(min_length=1)
    source_template: TemplateSource
    source_template_updated_at: datetime
    fixed_stage_sequence: tuple[StageType, ...]
    stage_role_bindings: tuple[StageRoleSnapshot, ...]
    approval_checkpoints: tuple[ApprovalType, ...]
    auto_regression_enabled: bool
    max_auto_regression_retries: int = Field(ge=0)
    schema_version: str = TEMPLATE_SNAPSHOT_SCHEMA_VERSION
    created_at: datetime

    @field_validator("fixed_stage_sequence")
    @classmethod
    def require_fixed_stage_sequence(
        cls,
        value: tuple[StageType, ...],
    ) -> tuple[StageType, ...]:
        if value != tuple(FIXED_STAGE_SEQUENCE):
            raise ValueError("fixed_stage_sequence must match the Function One stage order")
        return value

    @field_validator("stage_role_bindings")
    @classmethod
    def require_one_binding_per_stage(
        cls,
        value: tuple[StageRoleSnapshot, ...],
    ) -> tuple[StageRoleSnapshot, ...]:
        expected = tuple(FIXED_STAGE_SEQUENCE)
        actual = tuple(binding.stage_type for binding in value)
        if actual != expected:
            raise ValueError("stage_role_bindings must match fixed_stage_sequence")
        return value

    @field_validator("approval_checkpoints")
    @classmethod
    def require_fixed_approval_checkpoints(
        cls,
        value: tuple[ApprovalType, ...],
    ) -> tuple[ApprovalType, ...]:
        if value != tuple(FIXED_APPROVAL_CHECKPOINTS):
            raise ValueError("approval_checkpoints must match the Function One checkpoints")
        return value


class TemplateSnapshotBuilder:
    @staticmethod
    def build_for_run(
        template: Any,
        *,
        run_id: str,
        created_at: datetime,
    ) -> TemplateSnapshot:
        bindings = tuple(
            StageRoleSnapshot(
                stage_type=StageType(binding["stage_type"]),
                role_id=str(binding["role_id"]),
                system_prompt=str(binding["system_prompt"]).strip(),
                provider_id=str(binding["provider_id"]),
            )
            for binding in list(template.stage_role_bindings)
        )
        return TemplateSnapshot(
            snapshot_ref=f"template-snapshot-{run_id}",
            run_id=run_id,
            source_template_id=template.template_id,
            source_template_name=template.name,
            source_template=template.template_source,
            source_template_updated_at=template.updated_at,
            fixed_stage_sequence=tuple(
                StageType(stage) for stage in list(template.fixed_stage_sequence)
            ),
            stage_role_bindings=bindings,
            approval_checkpoints=tuple(
                ApprovalType(checkpoint)
                for checkpoint in list(template.approval_checkpoints)
            ),
            auto_regression_enabled=bool(template.auto_regression_enabled),
            max_auto_regression_retries=int(template.max_auto_regression_retries),
            created_at=created_at,
        )


__all__ = [
    "TEMPLATE_SNAPSHOT_SCHEMA_VERSION",
    "StageRoleSnapshot",
    "TemplateSnapshot",
    "TemplateSnapshotBuilder",
]
```

from __future__ import annotations

from collections.abc import Callable
from datetime import UTC, datetime

from sqlalchemy.orm import Session

from backend.app.db.models.runtime import PipelineRunModel
from backend.app.domain.template_snapshot import TemplateSnapshot


class RunLifecycleService:
    def __init__(
        self,
        session: Session,
        *,
        now: Callable[[], datetime] | None = None,
    ) -> None:
        self._session = session
        self._now = now or (lambda: datetime.now(UTC))

    def attach_template_snapshot(
        self,
        run: PipelineRunModel,
        snapshot: TemplateSnapshot,
    ) -> PipelineRunModel:
        if run.run_id != snapshot.run_id:
            raise ValueError("template snapshot run_id must match PipelineRun.run_id")
        run.template_snapshot_ref = snapshot.snapshot_ref
        run.updated_at = self._now()
        self._session.add(run)
        return run


__all__ = ["RunLifecycleService"]
```

## Review Hardening Applied

The final implementation intentionally hardens the initial minimal code in this plan:

- `TemplateSnapshot` and `StageRoleSnapshot` use strict scalar fields for frozen snapshot data.
- `schema_version` is constrained to the stable literal `template-snapshot-v1`.
- `TemplateSnapshotBuilder` rejects malformed JSON-backed template rows instead of coercing `None`, strings, or booleans into valid-looking snapshot values.
- Tests verify every stage slot's `stage_type`, `role_id`, `system_prompt`, and `provider_id`.
- The attach test persists and re-queries `PipelineRunModel` through a new runtime session.
