# L2.2 Redaction Payload Summarizer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` to implement this plan task-by-task. Fallback to `superpowers:executing-plans` only if subagents cannot be dispatched or the bounded file set below becomes unsafe. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the L2.2 `RedactionPolicy` foundation that redacts, blocks, summarizes, bounds, and hashes log/audit payloads before L2.3 JSONL writing and L2.4 audit recording consume them.

**Architecture:** `backend/app/observability/redaction.py` owns payload sanitization and summary generation. It consumes the existing observability contract enum `RedactionStatus`, adds the missing `unserializable` state required by L2.2, and returns immutable result objects containing only sanitized payload fragments, bounded excerpts, size metadata, content hashes, and structured summaries. It does not write JSONL files, update `log.db`, create audit records, or replace product facts.

**Tech Stack:** Python 3.11+, Pydantic enum contracts, dataclasses, standard-library `json`, `hashlib`, and `pytest` through `uv run --no-sync`.

```python
from __future__ import annotations

from collections.abc import Mapping, Sequence
from dataclasses import dataclass
from hashlib import sha256
import json
import re
from typing import Any

from backend.app.schemas.observability import RedactionStatus


_BLOCKED_FIELD_PLACEHOLDER = "[blocked:sensitive_field]"
_SENSITIVE_TEXT_PLACEHOLDER = "[blocked:sensitive_text_pattern]"
_UNSERIALIZABLE_PLACEHOLDER = "[blocked:payload_unserializable]"
_TRUNCATION_MARKER = "...[truncated]"

_SENSITIVE_KEY_TOKENS = frozenset(
    {
        "api_key",
        "authorization",
        "cookie",
        "credential",
        "password",
        "private_key",
        "secret",
        "token",
    }
)
_SAFE_REFERENCE_SUFFIXES = ("_ref", "_refs", "_status", "_id", "_ids")
_SENSITIVE_TEXT_PATTERNS = (
    re.compile(r"-----BEGIN [A-Z ]*PRIVATE KEY-----"),
    re.compile(r"\bBearer\s+[A-Za-z0-9._~+/=-]{12,}"),
    re.compile(r"\bsk-[A-Za-z0-9][A-Za-z0-9_-]{12,}"),
)


@dataclass(frozen=True)
class RedactedPayload:
    summary: dict[str, Any]
    excerpt: str
    redacted_payload: Any | None
    payload_size_bytes: int
    content_hash: str
    redaction_status: RedactionStatus


class RedactionPolicy:
    def __init__(
        self,
        *,
        max_text_length: int = 512,
        excerpt_length: int = 512,
    ) -> None:
        if max_text_length <= len(_TRUNCATION_MARKER):
            raise ValueError("max_text_length must be longer than truncation marker")
        if excerpt_length <= 0:
            raise ValueError("excerpt_length must be positive")
        self.max_text_length = max_text_length
        self.excerpt_length = excerpt_length

    def redact_mapping(
        self,
        payload: Mapping[str, Any],
        *,
        payload_type: str = "payload",
    ) -> RedactedPayload:
        return self.summarize_payload(payload, payload_type=payload_type)

    def summarize_payload(
        self,
        payload: Any,
        *,
        payload_type: str = "payload",
    ) -> RedactedPayload:
        if isinstance(payload, str):
            return self.summarize_text(payload, payload_type=payload_type)

        state = _RedactionState()
        try:
            sanitized = self._sanitize_value(payload, path="", state=state)
            original_size = _json_size(payload)
        except (TypeError, ValueError):
            return _blocked_result(
                payload_type=payload_type,
                reason="payload_unserializable",
                input_type=type(payload).__name__,
                status=RedactionStatus.UNSERIALIZABLE,
            )

        excerpt = _bounded_excerpt(sanitized, self.excerpt_length)
        status = _status_from_state(state)
        return RedactedPayload(
            summary={
                "payload_type": payload_type,
                "input_type": type(payload).__name__,
                "blocked_fields": state.blocked_fields,
                "truncated_fields": state.truncated_fields,
            },
            excerpt=excerpt,
            redacted_payload=sanitized,
            payload_size_bytes=original_size,
            content_hash=_hash_sanitized(sanitized),
            redaction_status=status,
        )

    def summarize_text(
        self,
        text: str,
        *,
        payload_type: str = "text",
    ) -> RedactedPayload:
        if _contains_sensitive_text(text):
            return _blocked_result(
                payload_type=payload_type,
                reason="sensitive_text_pattern",
                input_type="str",
                status=RedactionStatus.BLOCKED,
            )

        sanitized, was_truncated = self._bound_text(text)
        status = RedactionStatus.REDACTED if was_truncated else RedactionStatus.NOT_REQUIRED
        return RedactedPayload(
            summary={
                "payload_type": payload_type,
                "input_type": "str",
                "blocked_fields": [],
                "truncated_fields": ["$"] if was_truncated else [],
            },
            excerpt=sanitized,
            redacted_payload=sanitized,
            payload_size_bytes=len(text.encode("utf-8")),
            content_hash=_hash_sanitized(sanitized),
            redaction_status=status,
        )

    def _sanitize_value(
        self,
        value: Any,
        *,
        path: str,
        state: "_RedactionState",
    ) -> Any:
        if isinstance(value, Mapping):
            sanitized: dict[str, Any] = {}
            for raw_key, raw_value in value.items():
                if not isinstance(raw_key, str):
                    raise TypeError("mapping keys must be strings for log payloads")
                child_path = _join_path(path, raw_key)
                if _is_sensitive_key(raw_key):
                    sanitized[raw_key] = _BLOCKED_FIELD_PLACEHOLDER
                    state.blocked_fields.append(child_path)
                    continue
                sanitized[raw_key] = self._sanitize_value(
                    raw_value,
                    path=child_path,
                    state=state,
                )
            return sanitized

        if isinstance(value, Sequence) and not isinstance(value, (str, bytes, bytearray)):
            return [
                self._sanitize_value(item, path=f"{path}[{index}]", state=state)
                for index, item in enumerate(value)
            ]

        if isinstance(value, str):
            if _contains_sensitive_text(value):
                state.blocked_fields.append(path or "$")
                return _SENSITIVE_TEXT_PLACEHOLDER
            bounded, was_truncated = self._bound_text(value)
            if was_truncated:
                state.truncated_fields.append(path or "$")
            return bounded

        if value is None or isinstance(value, (bool, int, float)):
            return value

        raise TypeError(f"payload value at {path or '$'} is not JSON serializable")

    def _bound_text(self, text: str) -> tuple[str, bool]:
        if len(text) <= self.max_text_length:
            return text, False
        visible_length = self.max_text_length - len(_TRUNCATION_MARKER)
        return f"{text[:visible_length]}{_TRUNCATION_MARKER}", True


@dataclass
class _RedactionState:
    blocked_fields: list[str]
    truncated_fields: list[str]

    def __init__(self) -> None:
        self.blocked_fields = []
        self.truncated_fields = []


def _status_from_state(state: _RedactionState) -> RedactionStatus:
    if state.blocked_fields or state.truncated_fields:
        return RedactionStatus.REDACTED
    return RedactionStatus.NOT_REQUIRED


def _is_sensitive_key(key: str) -> bool:
    normalized = key.lower().replace("-", "_")
    if normalized.endswith(_SAFE_REFERENCE_SUFFIXES):
        return False
    parts = [part for part in normalized.split("_") if part]
    return normalized in _SENSITIVE_KEY_TOKENS or any(
        part in _SENSITIVE_KEY_TOKENS for part in parts
    )


def _contains_sensitive_text(text: str) -> bool:
    return any(pattern.search(text) for pattern in _SENSITIVE_TEXT_PATTERNS)


def _join_path(parent: str, key: str) -> str:
    return f"{parent}.{key}" if parent else key


def _json_size(payload: Any) -> int:
    return len(
        json.dumps(
            payload,
            ensure_ascii=False,
            separators=(",", ":"),
            sort_keys=True,
        ).encode("utf-8")
    )


def _bounded_excerpt(payload: Any, excerpt_length: int) -> str:
    encoded = json.dumps(
        payload,
        ensure_ascii=False,
        separators=(",", ":"),
        sort_keys=True,
    )
    if len(encoded) <= excerpt_length:
        return encoded
    visible_length = excerpt_length - len(_TRUNCATION_MARKER)
    if visible_length <= 0:
        return _TRUNCATION_MARKER
    return f"{encoded[:visible_length]}{_TRUNCATION_MARKER}"


def _hash_sanitized(payload: Any) -> str:
    encoded = json.dumps(
        payload,
        ensure_ascii=False,
        separators=(",", ":"),
        sort_keys=True,
    )
    return f"sha256:{sha256(encoded.encode('utf-8')).hexdigest()}"


def _blocked_result(
    *,
    payload_type: str,
    reason: str,
    input_type: str,
    status: RedactionStatus,
) -> RedactedPayload:
    summary = {
        "payload_type": payload_type,
        "blocked_reason": reason,
        "input_type": input_type,
    }
    placeholder = (
        _SENSITIVE_TEXT_PLACEHOLDER
        if reason == "sensitive_text_pattern"
        else _UNSERIALIZABLE_PLACEHOLDER
    )
    return RedactedPayload(
        summary=summary,
        excerpt=placeholder,
        redacted_payload=None,
        payload_size_bytes=0,
        content_hash=_hash_sanitized(summary),
        redaction_status=status,
    )


__all__ = ["RedactedPayload", "RedactionPolicy"]
```
