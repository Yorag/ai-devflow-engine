**Goal:** Implement pause, resume, and terminate frontend control entry points so runtime control stays separate from ordinary message sending, remains bound to the current active run, and matches the reviewed workspace interaction semantics.

**Architecture:** Keep the Composer main lifecycle button as the executable `发送` / `暂停` / `恢复` control surface, add one focused secondary Composer control component only for the `waiting_clarification + send` pause escape hatch, and add one workspace-toolbar terminate component for destructive termination. Wire all controls to the existing `frontend/src/api/runs.ts` commands with query invalidation instead of new store mutations. Keep all action eligibility derived from the current active run projection and existing Composer lifecycle resolver so history runs never surface executable controls.

**Tech Stack:** React 18, TypeScript strict mode, TanStack Query, Vitest, Testing Library, existing workspace projection types and workspace baseline CSS.

```tsx
import { useQueryClient } from "@tanstack/react-query";
import { useState } from "react";

import type { ApiRequestOptions } from "../../api/client";
import { apiQueryKeys } from "../../api/hooks";
import { pauseRun } from "../../api/runs";
import type { ComposerStateProjection } from "../../api/types";

type RunControlButtonsProps = {
  sessionId: string;
  runId: string | null;
  secondaryActions: ComposerStateProjection["secondary_actions"];
  isBusy: boolean;
  request?: ApiRequestOptions;
};

export function RunControlButtons({
  sessionId,
  runId,
  secondaryActions,
  isBusy,
  request,
}: RunControlButtonsProps): JSX.Element | null {
  const queryClient = useQueryClient();
  const [isSubmitting, setSubmitting] = useState(false);

  const canPause = canPauseRun(secondaryActions, runId);

  if (!canPause) {
    return null;
  }

  async function handlePause() {
    if (!runId || isBusy || isSubmitting) {
      return;
    }

    setSubmitting(true);
    try {
      await pauseRun(runId, request);
      await queryClient.invalidateQueries({
        queryKey: apiQueryKeys.sessionWorkspace(sessionId),
        refetchType: "all",
      });
    } finally {
      setSubmitting(false);
    }
  }

  return (
    <div className="composer__run-controls" aria-label="Current run controls">
      <button
        type="button"
        className="workspace-button workspace-button--secondary"
        disabled={isBusy || isSubmitting}
        onClick={handlePause}
        aria-label="暂停当前运行"
      >
        {isSubmitting ? "暂停中" : "暂停"}
      </button>
    </div>
  );
}

export function canPauseRun(
  secondaryActions: ComposerStateProjection["secondary_actions"],
  runId: string | null,
): boolean {
  if (!runId) {
    return false;
  }

  return secondaryActions.includes("pause");
}
```

Create `frontend/src/features/runs/TerminateRunAction.tsx`:

```tsx
import { useQueryClient } from "@tanstack/react-query";
import { useState } from "react";

import type { ApiRequestOptions } from "../../api/client";
import { apiQueryKeys } from "../../api/hooks";
import { terminateRun } from "../../api/runs";
import type { SessionStatus } from "../../api/types";

type TerminateRunActionProps = {
  sessionId: string;
  runId: string | null;
  sessionStatus: SessionStatus;
  request?: ApiRequestOptions;
};

export function TerminateRunAction({
  sessionId,
  runId,
  sessionStatus,
  request,
}: TerminateRunActionProps): JSX.Element | null {
  const queryClient = useQueryClient();
  const [isSubmitting, setSubmitting] = useState(false);

  if (!canTerminateRun(runId, sessionStatus)) {
    return null;
  }

  async function handleTerminate() {
    if (!runId || isSubmitting) {
      return;
    }

    const confirmed = window.confirm("终止当前运行后将保留历史记录。继续吗？");
    if (!confirmed) {
      return;
    }

    setSubmitting(true);
    try {
      await terminateRun(runId, request);
      await queryClient.invalidateQueries({
        queryKey: apiQueryKeys.sessionWorkspace(sessionId),
        refetchType: "all",
      });
    } finally {
      setSubmitting(false);
    }
  }

  return (
    <button
      type="button"
      className="workspace-button workspace-button--danger workspace-button--compact"
      onClick={handleTerminate}
      disabled={isSubmitting}
      aria-label="终止当前运行"
    >
      {isSubmitting ? "终止中" : "终止"}
    </button>
  );
}

export function canTerminateRun(
  runId: string | null,
  sessionStatus: SessionStatus,
): boolean {
  if (!runId) {
    return false;
  }

  return !["draft", "completed", "failed", "terminated"].includes(sessionStatus);
}
```
