**Goal:** Implement `PromptValidation` so editable `system_prompt` values are rejected before template save and before run startup whenever they try to override platform boundaries, stage contracts, tool limits, approval semantics, delivery semantics, audit semantics, structured output requirements, or prompt budget limits.

**Architecture:** Add a runtime-owned `PromptValidationService` that consumes the same stage-contract source as `GraphCompiler`, validates trimmed prompt text against fixed deny rules plus prompt-length/context-budget limits derived from current runtime settings, and exposes two integration entrypoints: one for template save in `TemplateService` and one for run-start template snapshots in `RunLifecycleService`. Keep template-save and run-start hook surfaces unified through this service rather than duplicating validation logic or adding a second stage-rule table.

**Tech Stack:** Python 3.11+, Pydantic v2 models already in the repo, existing `GraphCompiler` stage-contract source, existing control/runtime services, and pytest through `uv run pytest`.

## Design Decisions

- `PromptValidationService` owns three public methods only:
  - `validate_system_prompt(...)`
  - `validate_template_prompts_before_save(...)`
  - `validate_run_prompt_snapshots(...)`
- The service must derive stage constraints from the same source used by `GraphCompiler._build_stage_contracts(...)`, not from a duplicate static table. The implementation may instantiate `GraphCompiler` with a runtime-limit snapshot built from current settings and a template-shaped input to obtain the canonical stage contracts.
- Prompt rejection remains a stable validation-class error. Template-save rejections continue to surface as `TemplateServiceError(ErrorCode.VALIDATION_ERROR, ..., 422 or existing save-path status)` and run-start rejections continue to surface as `RunPromptValidationError` mapped to `422`.
- Length/context budget enforcement uses current runtime settings through `PlatformRuntimeSettingsRead.context_limits` plus a fixed prompt max-length guard owned by this slice. Exceeding limits must reject rather than truncate.
- Validation examines explicit policy-bypass instructions in prompt text and stage-specific conflicts such as instructing disallowed tools or disabling structured outputs. It must not judge writing style, architecture taste, or solution quality.
- Template-save rejections must still record the existing rejected audit path for template commands. Run-start validation failures must still record the existing startup rejected path and runtime log path.

```python
from __future__ import annotations

from dataclasses import dataclass
from typing import Iterable

from backend.app.api.error_codes import ErrorCode
from backend.app.domain.enums import StageType
from backend.app.domain.runtime_limit_snapshot import RuntimeLimitSnapshotBuilder
from backend.app.domain.template_snapshot import TemplateSnapshot
from backend.app.schemas.runtime_settings import PlatformRuntimeSettingsRead
from backend.app.services.graph_compiler import GraphCompiler


@dataclass(frozen=True, slots=True)
class PromptValidationRule:
    rule_id: str
    message: str


@dataclass(frozen=True, slots=True)
class PromptValidationResult:
    accepted: bool
    normalized_prompt: str
    error_code: ErrorCode | None
    rule_ids: list[str]
    message: str | None = None


class PromptValidationError(ValueError):
    def __init__(
        self,
        message: str,
        *,
        rule_ids: Iterable[str],
        error_code: ErrorCode = ErrorCode.VALIDATION_ERROR,
    ) -> None:
        self.message = message
        self.rule_ids = list(rule_ids)
        self.error_code = error_code
        super().__init__(message)


class PromptValidationService:
    def __init__(
        self,
        *,
        settings_read: PlatformRuntimeSettingsRead,
        max_prompt_chars: int = 12000,
    ) -> None:
        self._settings_read = settings_read
        self._max_prompt_chars = max_prompt_chars
        self._stage_contracts = self._load_stage_contracts()

    def validate_system_prompt(
        self,
        *,
        prompt_text: str,
        stage_type: StageType,
    ) -> PromptValidationResult:
        normalized = prompt_text.strip()
        rule_ids: list[str] = []
        lowered = normalized.lower()
        if not normalized:
            rule_ids.append("blank_prompt")
        if len(normalized) > self._max_prompt_chars:
            rule_ids.append("prompt_length_exceeded")
        if "ignore platform instruction" in lowered or "ignore platform instructions" in lowered:
            rule_ids.append("platform_instruction_override")
        if "any tool you want" in lowered or "unauthorized tool" in lowered:
            rule_ids.append("unauthorized_tool_override")
        if "skip approval" in lowered:
            rule_ids.append("approval_boundary_override")
        if "disable structured output" in lowered or "free-form prose only" in lowered:
            rule_ids.append("structured_output_override")
        if "raw chain-of-thought" in lowered:
            rule_ids.append("chain_of_thought_leak")
        if "change the delivery mode" in lowered:
            rule_ids.append("delivery_boundary_override")
        if "do not log or audit" in lowered:
            rule_ids.append("audit_boundary_override")
        if self._mentions_disallowed_tools(lowered=lowered, stage_type=stage_type):
            rule_ids.append("stage_contract_tool_override")
        if rule_ids:
            raise PromptValidationError(
                "System prompt conflicts with platform prompt validation rules.",
                rule_ids=rule_ids,
            )
        return PromptValidationResult(
            accepted=True,
            normalized_prompt=normalized,
            error_code=None,
            rule_ids=[],
        )

    def validate_template_prompts_before_save(
        self,
        bindings: list[dict[str, str]],
    ) -> list[dict[str, str]]:
        validated: list[dict[str, str]] = []
        for binding in bindings:
            stage_type = StageType(binding["stage_type"])
            result = self.validate_system_prompt(
                prompt_text=binding["system_prompt"],
                stage_type=stage_type,
            )
            validated.append({**binding, "system_prompt": result.normalized_prompt})
        return validated

    def validate_run_prompt_snapshots(
        self,
        *,
        template_snapshot: TemplateSnapshot,
    ) -> None:
        for binding in template_snapshot.stage_role_bindings:
            self.validate_system_prompt(
                prompt_text=binding.system_prompt,
                stage_type=binding.stage_type,
            )

    def _mentions_disallowed_tools(
        self,
        *,
        lowered: str,
        stage_type: StageType,
    ) -> bool:
        contract = self._stage_contracts[stage_type.value]
        allowed_tools = set(contract["allowed_tools"])
        if "bash" in lowered and "bash" not in allowed_tools:
            return True
        if "write_file" in lowered and "write_file" not in allowed_tools:
            return True
        if "edit_file" in lowered and "edit_file" not in allowed_tools:
            return True
        return False

    def _load_stage_contracts(self) -> dict[str, dict[str, object]]:
        runtime_limit_snapshot = RuntimeLimitSnapshotBuilder.build_for_run(
            self._settings_read,
            template_snapshot=self._template_snapshot_stub(),
            run_id="prompt-validation-stage-contracts",
            created_at=self._settings_read.version.updated_at,
        )
        return GraphCompiler(now=lambda: self._settings_read.version.updated_at).compile(
            template_snapshot=self._template_snapshot_stub(),
            runtime_limit_snapshot=runtime_limit_snapshot,
        ).stage_contracts

    def _template_snapshot_stub(self) -> TemplateSnapshot:
        ...
```
