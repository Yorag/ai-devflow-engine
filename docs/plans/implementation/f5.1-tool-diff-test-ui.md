```tsx
import type { StageItemProjection } from "../../api/types";

type ToolCallParseResult = {
  toolName: string;
  targetSummary: string;
  statusLabel: string;
  durationLabel: string;
  commandExcerpt: string;
  outputSummary: string;
};

export function ToolCallItem({ item }: { item: StageItemProjection }): JSX.Element {
  const parsed = parseToolCallContent(item.content);

  return (
    <li className="stage-node-item stage-node-item--tool-call" aria-label="Tool Call stage item">
      <header className="stage-node-item__header">
        <span>Tool Call</span>
        <strong>{item.title}</strong>
        <time dateTime={item.occurred_at}>{formatTimestamp(item.occurred_at)}</time>
      </header>
      {item.summary ? <p className="stage-node-item__summary">{item.summary}</p> : null}
      <div className="stage-node-item__tool-grid" aria-label="Tool call metadata">
        <ToolDatum label="Tool" value={parsed.toolName} />
        <ToolDatum label="Target" value={parsed.targetSummary} />
        <ToolDatum label="Status" value={parsed.statusLabel} />
        <ToolDatum label="Duration" value={parsed.durationLabel} />
      </div>
      <p className="stage-node-item__command">{parsed.commandExcerpt}</p>
      <details className="stage-node-item__details">
        <summary>{parsed.outputSummary}</summary>
        <pre>{item.content}</pre>
      </details>
      <ReferenceList refs={item.artifact_refs} />
    </li>
  );
}

function ToolDatum({ label, value }: { label: string; value: string }): JSX.Element {
  return (
    <span className="stage-node-item__datum">
      <strong>{label}</strong>
      <span>{value}</span>
    </span>
  );
}

function ReferenceList({ refs }: { refs: string[] }): JSX.Element | null {
  if (refs.length === 0) {
    return null;
  }

  return (
    <div className="stage-node-item__refs" aria-label="Artifact references">
      {refs.map((ref) => (
        <span key={ref}>{ref}</span>
      ))}
    </div>
  );
}

function parseToolCallContent(content: string | null): ToolCallParseResult {
  const lines = content?.split("\n") ?? [];
  const commandExcerpt = lines[0] ?? "Tool call";
  const toolName = commandExcerpt.split(" ")[0] ?? "tool";
  const targetSummary =
    lines.find((line) => line.startsWith("Target:"))?.replace("Target:", "").trim() ??
    "Not recorded";
  const statusLabel = formatStatusLabel(
    lines.find((line) => line.startsWith("Status:"))?.replace("Status:", "").trim() ??
      "unknown",
  );
  const outputSummary =
    lines
      .find((line) => line.startsWith("Output summary:"))
      ?.replace("Output summary:", "")
      .trim() ?? "Command output";

  return {
    toolName,
    targetSummary,
    statusLabel,
    durationLabel: "Not recorded",
    commandExcerpt,
    outputSummary,
  };
}

function formatDuration(value: number | null): string {
  if (typeof value !== "number") {
    return "Not recorded";
  }
  if (value >= 1000) {
    return `${Number((value / 1000).toFixed(1))}s`;
  }
  return `${value}ms`;
}

function formatStatusLabel(value: string): string {
  return value
    .split("_")
    .map((part) => part.charAt(0).toUpperCase() + part.slice(1))
    .join(" ");
}

function formatTimestamp(value: string): string {
  return value.includes("T") ? value.replace("T", " ").slice(0, 16) : value;
}
```

```tsx
import type { StageItemProjection } from "../../api/types";

type DiffPreviewParseResult = {
  files: string[];
  preview: string;
  remainder: string | null;
};

export function DiffPreview({ item }: { item: StageItemProjection }): JSX.Element {
  const parsed = parseDiffPreviewContent(item.content);

  return (
    <li className="stage-node-item stage-node-item--diff-preview" aria-label="Diff Preview stage item">
      <header className="stage-node-item__header">
        <span>Diff Preview</span>
        <strong>{item.title}</strong>
        <time dateTime={item.occurred_at}>{formatTimestamp(item.occurred_at)}</time>
      </header>
      {item.summary ? <p className="stage-node-item__summary">{item.summary}</p> : null}
      <ul className="stage-node-item__file-list" aria-label="Changed files">
        {parsed.files.map((file) => (
          <li key={file}>{file}</li>
        ))}
      </ul>
      <pre className="stage-node-item__diff-snippet">{parsed.preview}</pre>
      {parsed.remainder ? (
        <details className="stage-node-item__details">
          <summary>More diff context</summary>
          <pre>{parsed.remainder}</pre>
        </details>
      ) : null}
      <ReferenceList refs={item.artifact_refs} />
    </li>
  );
}

function parseDiffPreviewContent(content: string | null): DiffPreviewParseResult {
  const sections = content?.split("\n\n") ?? [];
  const fileSection = sections[0] ?? "";
  const preview = sections[1] ?? "";
  const remainder = sections.slice(2).join("\n\n") || null;
  const files = fileSection
    .replace(/^Files:\n?/u, "")
    .split("\n")
    .map((line) => line.trim())
    .filter(Boolean);

  return {
    files,
    preview,
    remainder,
  };
}

function ReferenceList({ refs }: { refs: string[] }): JSX.Element | null {
  if (refs.length === 0) {
    return null;
```
