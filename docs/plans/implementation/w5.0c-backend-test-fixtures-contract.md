@dataclass(slots=True)
class FakeProviderError(RuntimeError):
    failure_kind: str
    message: str
    error_code: ErrorCode | None = None

    def __post_init__(self) -> None:
        RuntimeError.__init__(self, self.message)


@dataclass
class FakeChatModel:
    _script: deque[dict[str, Any]] = field(default_factory=deque)

    def enqueue(self, item: dict[str, Any]) -> None:
        self._script.append(item)

    def invoke_structured(self) -> dict[str, Any]:
        if not self._script:
            raise FakeProviderError("no_scripted_response", "No scripted provider response is available.")
        item = self._script.popleft()
        kind = item["kind"]
        if kind == "structured_success":
            return item["payload"]
        if kind == "tool_call_request":
            return {
                "decision_type": "request_tool_call",
                "tool_name": item["tool_name"],
                "input_payload": item["input_payload"],
            }
        if kind == "structured_failure":
            raise FakeProviderError(
                "structured_failure",
                item["message"],
                error_code=item.get("error_code"),
            )
        raise FakeProviderError(kind, item["message"])


@dataclass
class FakeProvider:
    provider_snapshot: ProviderSnapshotRead
    model_binding_snapshot: ModelBindingSnapshotRead
    config: ProviderConfig
    chat_model: FakeChatModel = field(default_factory=FakeChatModel)

    def enqueue_structured_success(self, payload: dict[str, Any]) -> None:
        self.chat_model.enqueue({"kind": "structured_success", "payload": payload})

    def enqueue_tool_call_request(self, tool_name: str, input_payload: dict[str, Any]) -> None:
        self.chat_model.enqueue(
            {
                "kind": "tool_call_request",
                "tool_name": tool_name,
                "input_payload": input_payload,
            }
        )

    def enqueue_structured_failure(
        self,
        *,
        error_code: ErrorCode | None,
        message: str,
    ) -> None:
        self.chat_model.enqueue(
            {
                "kind": "structured_failure",
                "error_code": error_code,
                "message": message,
            }
        )

    def enqueue_timeout(self) -> None:
        self.chat_model.enqueue({"kind": "timeout", "message": "Provider request timed out."})

    def enqueue_rate_limit(self) -> None:
        self.chat_model.enqueue({"kind": "rate_limit", "message": "Provider rate limit exceeded."})

    def enqueue_network_error(self) -> None:
        self.chat_model.enqueue({"kind": "network_error", "message": "Provider network request failed."})
