## Key Behavior Decisions

- The tool input contract is intentionally minimal and explicit:
  - `pattern`: required regex string passed to `rg`
  - `path`: required workspace-relative search root; callers that want full-workspace search pass `"."`
- The tool result contract is:
  - `matches`: ordered list of `{"path", "line_number", "snippet", "snippet_truncated"}`
  - `truncated`: `true` when matches were cut off by the configured max-result boundary
- `WorkspaceGrepOptions` owns execution-time policy local to this slice:
  - `executable = "rg"`
  - `max_results = 100`
  - `snippet_char_limit = 240`
  - excluded globs for `.runtime/logs`, dependency directories, and common build outputs
- `GrepTool` remains read-only for risk classification, but it must participate in audit recording for permission-denied and sensitive-content-blocked failures. The least invasive way to do that inside the existing W5.0b stack is to set `side_effect_level = ToolSideEffectLevel.WORKSPACE_READ`, which makes the existing audit-intent path run without inventing a second audit channel.
- `rg` availability is checked at execution time before calling `subprocess.run`. Missing `rg` returns a structured tool failure using the existing catalog and a stable safe-details reason such as `rg_unavailable`; it must not silently fall back to Python file scanning.
- `grep()` uses `subprocess.run([...], shell=False, cwd=workspace.root, ...)` only. No shell command string, no `bash`, and no alternate grep implementation are allowed.
- `rg` JSON output is parsed from `--json`. Only `match` events become payload rows.
- Sensitive match content is blocked before it reaches the final tool payload:
  - if a candidate snippet trips `RedactionPolicy.summarize_text(...).redaction_status == BLOCKED`, the whole tool call fails with a structured safe-details reason such as `grep_match_blocked`
  - the final error must not echo the secret-bearing line text
- Snippet truncation is a grep-local presentation rule, not the same as the large-preview redaction helper. The payload snippet should keep a short visible prefix and append `...`, while the final tool `output_preview` still goes through the existing `_preview()` helper.
- Permission-denied or other non-timeout `rg` execution failures return structured failures without copying raw stderr into safe details or previews.
- Result ordering is deterministic by `(path, line_number, snippet)` after parsing.

```python
def _build_harness(
    tmp_path: Path,
    *,
    options: WorkspaceGrepOptions | None = None,

    project_root = tmp_path / "project"
    project_root.mkdir()
    manager = WorkspaceManager(
        settings=EnvironmentSettings(
            default_project_root=project_root,
            workspace_root=project_root,

    run_log = _RecordingRunLog()

        workspace=workspace,
        registry=registry,
        trace_context=trace_context,
```

```python
class GrepTool:
    manager: WorkspaceManager
    workspace: RunWorkspace
    options: WorkspaceGrepOptions = WorkspaceGrepOptions()

    @property
    def name(self) -> str:
        return "grep"

    @property
    def category(self) -> str:
        return "workspace"

    @property
    def description(self) -> str:
        return "Search workspace text content with local ripgrep."

    @property
    def input_schema(self) -> Mapping[str, object]:
        return _GREP_INPUT_SCHEMA

    @property
    def result_schema(self) -> Mapping[str, object]:
        return _GREP_RESULT_SCHEMA

    @property
    def default_risk_level(self) -> ToolRiskLevel:
        return ToolRiskLevel.READ_ONLY

    @property
    def risk_categories(self) -> Sequence[ToolRiskCategory]:
        return ()

    @property
    def permission_boundary(self) -> ToolPermissionBoundary:
        return ToolPermissionBoundary(
            boundary_type="workspace",
            requires_workspace=True,
            resource_scopes=("current_run_workspace",),
            workspace_target_paths=("path",),
        )

    @property
    def side_effect_level(self) -> ToolSideEffectLevel:
        return ToolSideEffectLevel.WORKSPACE_READ

    @property
    def audit_required(self) -> bool:
        return False

    @property
    def schema_version(self) -> str:
        return "tool-schema-v1"

    @property
    def default_timeout_seconds(self) -> float | None:
        return 5.0

    def bindable_description(self) -> ToolBindableDescription:
        return ToolBindableDescription(
            name=self.name,
            description=self.description,
            input_schema=dict(self.input_schema),
            result_schema=dict(self.result_schema),
            risk_level=self.default_risk_level,
            risk_categories=list(self.risk_categories),
            schema_version=self.schema_version,
            default_timeout_seconds=self.default_timeout_seconds,
        )

    def execute(self, tool_input: ToolInput) -> ToolResult:
        return grep(
            self.manager,
            self.workspace,
            str(tool_input.input_payload["pattern"]),
            str(tool_input.input_payload["path"]),

def grep(
    manager: WorkspaceManager,
    workspace: RunWorkspace,
    pattern: str,
    path: str,
    *,
    options: WorkspaceGrepOptions,
    tool_input: ToolInput,
) -> ToolResult:
    try:
        resolved = manager.assert_inside_workspace(
            path,
            workspace=workspace,
            trace_context=tool_input.trace_context,
        )
    except ToolWorkspaceBoundaryError as exc:
        return _workspace_boundary_failed_result(
            tool_input,
            tool_name="grep",
            target=exc.target,
        )

    normalized_path = _relative_path(workspace, resolved)
    executable = shutil.which(options.executable)
    if executable is None:
        return _tool_error_result(
            tool_input,
            tool_name="grep",
            error_code=ErrorCode.INTERNAL_ERROR,
            safe_details={"path": normalized_path or ".", "reason": "rg_unavailable"},
        )

    command = [
        executable,
        "--json",
        "--line-number",
        "--color",
        "never",
        "--hidden",
        "--no-messages",
    ]
    for glob_pattern in options.excluded_globs:
        command.extend(["-g", f"!{glob_pattern}"])
    command.extend([pattern, normalized_path or "."])

    try:
        completed = subprocess.run(
            command,
            cwd=workspace.root,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
            encoding="utf-8",
            errors="replace",
            timeout=tool_input.timeout_seconds,
            check=False,
        )
    except subprocess.TimeoutExpired:
        return _tool_error_result(
            tool_input,
            tool_name="grep",
            error_code=ErrorCode.TOOL_TIMEOUT,
            safe_details={"timeout_seconds": tool_input.timeout_seconds},
        )

    if completed.returncode not in (0, 1):
        reason = "rg_permission_denied" if "permission denied" in completed.stderr.lower() else "rg_failed"
        return _tool_error_result(
            tool_input,
            tool_name="grep",
            error_code=ErrorCode.INTERNAL_ERROR,
            safe_details={"path": normalized_path or ".", "reason": reason, "returncode": completed.returncode},
        )

    matches: list[GrepResultItem] = []
    truncated = False
    for raw_line in completed.stdout.splitlines():
        if not raw_line.strip():
            continue
        event = json.loads(raw_line)
        if event.get("type") != "match":
            continue
        data = event["data"]
        relative_path = str(data["path"]["text"]).replace("\\\\", "/")
        if _is_excluded_path(relative_path, workspace):
            continue
        line_number = int(data["line_number"])
        snippet, snippet_truncated, blocked = _grep_snippet(
            str(data["lines"]["text"]).rstrip("\\r\\n"),
            limit=options.snippet_char_limit,
        )
        if blocked:
            return _tool_error_result(
                tool_input,
                tool_name="grep",
                error_code=ErrorCode.INTERNAL_ERROR,
                safe_details={"path": relative_path, "line_number": line_number, "reason": "grep_match_blocked"},
            )
        if len(matches) >= options.max_results:
            truncated = True
            break
        matches.append(
            GrepResultItem(
                path=relative_path,
                line_number=line_number,
                snippet=snippet,
                snippet_truncated=snippet_truncated,
            )
        )

    payload_matches = [
        item.as_payload()
        for item in sorted(matches, key=lambda item: (item.path, item.line_number, item.snippet))
    ]
    return ToolResult(
        tool_name="grep",
        call_id=tool_input.call_id,
        status=ToolResultStatus.SUCCEEDED,
        output_payload={"matches": payload_matches, "truncated": truncated},
        output_preview=_preview(
            "\\n".join(
                f"{item['path']}:{item['line_number']}:{item['snippet']}"
                for item in payload_matches
            )
        ),
        trace_context=tool_input.trace_context,
        coordination_key=tool_input.coordination_key,
    )
```

```python
def _grep_snippet(text: str, *, limit: int) -> tuple[str, bool, bool]:
    redacted = RedactionPolicy(
        max_text_length=max(limit, 32),
        excerpt_length=max(limit, 32),
    ).summarize_text(
        text,
        payload_type="grep_match",
    )
    if redacted.redaction_status.value == "blocked":
        return "", False, True
    visible = str(redacted.redacted_payload or "")
    if len(visible) <= limit:
        return visible, False, False
    if limit <= 3:
        return "." * limit, True, False
    return f"{visible[: limit - 3]}...", True, False
```
