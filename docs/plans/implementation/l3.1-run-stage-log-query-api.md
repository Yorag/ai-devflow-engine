    _seed_log_rows(manager)

        manager.session(DatabaseRole.RUNTIME) as runtime_session,
        manager.session(DatabaseRole.LOG) as log_session,

        page = service.list_run_logs("run-active", limit=2)

        assert [entry.log_id for entry in page.entries] == ["log-run-1", "log-run-2"]
        assert page.has_more is True
        assert page.next_cursor == service.encode_cursor(LOG_TIMES[1], "log-run-2")
        assert page.query.run_id == "run-active"
        assert page.query.stage_run_id is None
        assert page.query.limit == 2

        filtered = service.list_run_logs(
            "run-active",
            level=LogLevel.ERROR,
            category=LogCategory.TOOL,
            source="tool.registry",
            since=LOG_TIMES[1],
            until=LOG_TIMES[2],
            limit=5,
        )
        assert [entry.log_id for entry in filtered.entries] == ["log-run-2"]
        assert filtered.has_more is False
        assert filtered.next_cursor is None

        second_page = service.list_run_logs(
            "run-active",
            cursor=page.next_cursor,
            limit=2,
        )
        assert [entry.log_id for entry in second_page.entries] == ["log-run-3"]
        assert second_page.has_more is False

import base64
import json
from datetime import UTC, datetime

from sqlalchemy import and_, or_, select
from sqlalchemy.orm import Session

from backend.app.api.error_codes import ErrorCode
from backend.app.db.models.control import PlatformRuntimeSettingsModel, ProjectModel, SessionModel
from backend.app.db.models.log import RunLogEntryModel
from backend.app.db.models.runtime import PipelineRunModel, StageRunModel
from backend.app.schemas.observability import (
    LogCategory,
    LogLevel,
    RedactionStatus,
    RunLogEntryProjection,
    RunLogQuery,
    RunLogQueryResponse,
)
from backend.app.schemas.runtime_settings import LogPolicy


SETTINGS_UNAVAILABLE_MESSAGE = "Current PlatformRuntimeSettings are unavailable."
RUN_LOG_NOT_FOUND_MESSAGE = "Run logs were not found."
STAGE_LOG_NOT_FOUND_MESSAGE = "Stage logs were not found."
LOG_QUERY_INVALID_MESSAGE = "Log query is invalid."


class LogQueryServiceError(RuntimeError):
    def __init__(self, error_code: ErrorCode, message: str, status_code: int) -> None:
        self.error_code = error_code
        self.message = message
        self.status_code = status_code
        super().__init__(message)


class LogQueryService:
    def __init__(
        self,
        control_session: Session,
        runtime_session: Session,
        log_session: Session,
    ) -> None:
        self._control_session = control_session
        self._runtime_session = runtime_session
        self._log_session = log_session

    def list_run_logs(
        self,
        run_id: str,
        *,
        level: LogLevel | None = None,
        category: LogCategory | None = None,
        source: str | None = None,
        since: datetime | None = None,
        until: datetime | None = None,
        cursor: str | None = None,
        limit: int | None = None,
    ) -> RunLogQueryResponse:
        run = self._visible_run(run_id)
        query = self._validated_query(
            run_id=run.run_id,
            stage_run_id=None,
            level=level,
            category=category,
            source=source,
            since=since,
            until=until,
            cursor=cursor,
            limit=limit,
        )
        return self._list_logs(query=query)

    def list_stage_logs(
        self,
        stage_run_id: str,
        *,
        level: LogLevel | None = None,
        category: LogCategory | None = None,
        source: str | None = None,
        since: datetime | None = None,
        until: datetime | None = None,
        cursor: str | None = None,
        limit: int | None = None,
    ) -> RunLogQueryResponse:
        stage = self._visible_stage(stage_run_id)
        query = self._validated_query(
            run_id=stage.run_id,
            stage_run_id=stage.stage_run_id,
            level=level,
            category=category,
            source=source,
            since=since,
            until=until,
            cursor=cursor,
            limit=limit,
        )
        return self._list_logs(query=query)

    def encode_cursor(self, created_at: datetime, log_id: str) -> str:
        normalized = created_at if created_at.tzinfo is not None else created_at.replace(tzinfo=UTC)
        payload = json.dumps(
            {"created_at": normalized.isoformat(), "log_id": log_id},
            separators=(",", ":"),
            sort_keys=True,
        ).encode("utf-8")
        return base64.urlsafe_b64encode(payload).decode("ascii")

    def decode_cursor(self, cursor: str) -> tuple[datetime, str]:
        try:
            decoded = base64.urlsafe_b64decode(cursor.encode("ascii"))
            payload = json.loads(decoded.decode("utf-8"))
            created_at = datetime.fromisoformat(payload["created_at"])
            log_id = payload["log_id"]
        except Exception as exc:
            raise LogQueryServiceError(
                ErrorCode.LOG_QUERY_INVALID,
                LOG_QUERY_INVALID_MESSAGE,
                422,
            ) from exc
        if not isinstance(log_id, str) or not log_id:
            raise LogQueryServiceError(
                ErrorCode.LOG_QUERY_INVALID,
                LOG_QUERY_INVALID_MESSAGE,
                422,
            )
        if created_at.tzinfo is None:
            created_at = created_at.replace(tzinfo=UTC)
        return created_at, log_id

    def _validated_query(self, **payload) -> RunLogQuery:
        default_limit, max_limit = self._log_query_limits()
        requested_limit = payload.pop("limit", None)
        resolved_limit = default_limit if requested_limit is None else requested_limit
        if resolved_limit <= 0 or resolved_limit > max_limit:
            raise LogQueryServiceError(
                ErrorCode.LOG_QUERY_INVALID,
                LOG_QUERY_INVALID_MESSAGE,
                422,
            )
        try:
            return RunLogQuery(limit=resolved_limit, **payload)
        except ValueError as exc:
            raise LogQueryServiceError(
                ErrorCode.LOG_QUERY_INVALID,
                LOG_QUERY_INVALID_MESSAGE,
                422,
            ) from exc

    def _list_logs(self, *, query: RunLogQuery) -> RunLogQueryResponse:
        statement = select(RunLogEntryModel).order_by(
            RunLogEntryModel.created_at.asc(),
            RunLogEntryModel.log_id.asc(),
        )
        if query.run_id is not None:
            statement = statement.where(RunLogEntryModel.run_id == query.run_id)
        if query.stage_run_id is not None:
            statement = statement.where(RunLogEntryModel.stage_run_id == query.stage_run_id)
        if query.level is not None:
            statement = statement.where(RunLogEntryModel.level == query.level)
        if query.category is not None:
            statement = statement.where(RunLogEntryModel.category == query.category)
        if query.source is not None:
            statement = statement.where(RunLogEntryModel.source == query.source)
        if query.since is not None:
            statement = statement.where(RunLogEntryModel.created_at >= query.since)
        if query.until is not None:
            statement = statement.where(RunLogEntryModel.created_at <= query.until)
        if query.cursor is not None:
            cursor_created_at, cursor_log_id = self.decode_cursor(query.cursor)
            statement = statement.where(
                or_(
                    RunLogEntryModel.created_at > cursor_created_at,
                    and_(
                        RunLogEntryModel.created_at == cursor_created_at,
                        RunLogEntryModel.log_id > cursor_log_id,
                    ),
                )
            )
        rows = self._log_session.execute(statement.limit(query.limit + 1)).scalars().all()
        page_rows = rows[: query.limit]
        has_more = len(rows) > query.limit
        next_cursor = None
        if has_more and page_rows:
            last = page_rows[-1]
            next_cursor = self.encode_cursor(last.created_at, last.log_id)
        return RunLogQueryResponse(
            entries=[self._projection(row) for row in page_rows],
            next_cursor=next_cursor,
            has_more=has_more,
            query=query,
        )

    def _projection(self, row: RunLogEntryModel) -> RunLogEntryProjection:
        created_at = row.created_at if row.created_at.tzinfo is not None else row.created_at.replace(tzinfo=UTC)
        return RunLogEntryProjection(
            log_id=row.log_id,
            session_id=row.session_id,
            run_id=row.run_id,
            stage_run_id=row.stage_run_id,
            approval_id=row.approval_id,
            tool_confirmation_id=row.tool_confirmation_id,
            delivery_record_id=row.delivery_record_id,
            graph_thread_id=row.graph_thread_id,
            request_id=row.request_id,
            source=row.source,
            category=row.category,
            level=row.level,
            message=row.message,
            log_file_ref=row.log_file_ref,
            line_offset=row.line_offset,
            line_number=row.line_number,
            log_file_generation=row.log_file_generation,
            payload_ref=row.payload_ref,
            payload_excerpt=row.payload_excerpt,
            payload_size_bytes=row.payload_size_bytes,
            redaction_status=row.redaction_status,
            correlation_id=row.correlation_id,
            trace_id=row.trace_id,
            span_id=row.span_id,
            parent_span_id=row.parent_span_id,
            created_at=created_at,
        )

    def _log_query_limits(self) -> tuple[int, int]:
        settings = self._control_session.get(
            PlatformRuntimeSettingsModel,
            "platform-runtime-settings",
        )
        if settings is None:
            raise LogQueryServiceError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                SETTINGS_UNAVAILABLE_MESSAGE,
                503,
            )
        try:
            policy = LogPolicy.model_validate(settings.log_policy)
        except Exception as exc:
            raise LogQueryServiceError(
                ErrorCode.CONFIG_SNAPSHOT_UNAVAILABLE,
                SETTINGS_UNAVAILABLE_MESSAGE,
                503,
            ) from exc
        return policy.log_query_default_limit, policy.log_query_max_limit

    def _visible_run(self, run_id: str) -> PipelineRunModel:
        run = self._runtime_session.get(PipelineRunModel, run_id)
        if run is None:
            raise LogQueryServiceError(ErrorCode.NOT_FOUND, RUN_LOG_NOT_FOUND_MESSAGE, 404)
        session = self._control_session.get(SessionModel, run.session_id)
        project = self._control_session.get(ProjectModel, run.project_id)
        if session is None or project is None or not session.is_visible or not project.is_visible:
            raise LogQueryServiceError(ErrorCode.NOT_FOUND, RUN_LOG_NOT_FOUND_MESSAGE, 404)
        if session.project_id != project.project_id:
            raise LogQueryServiceError(ErrorCode.NOT_FOUND, RUN_LOG_NOT_FOUND_MESSAGE, 404)
        return run

    def _visible_stage(self, stage_run_id: str) -> StageRunModel:
        stage = self._runtime_session.get(StageRunModel, stage_run_id)
        if stage is None:
            raise LogQueryServiceError(ErrorCode.NOT_FOUND, STAGE_LOG_NOT_FOUND_MESSAGE, 404)
        self._visible_run(stage.run_id)
        return stage
