class ContextTokenEstimator:
    def estimate_text(self, text: str) -> int:
        return max(1, (len(text) + 3) // 4) if text else 0

    def estimate_block(self, block: ContextBlock) -> int:
        if block.estimated_tokens is not None:
            return block.estimated_tokens
        return self.estimate_text(block.summary)

    def estimate_blocks(self, blocks: Sequence[ContextBlock]) -> int:
        return sum(self.estimate_block(block) for block in blocks)

    def estimate_envelope(self, envelope: ContextEnvelope) -> int:
        return self.estimate_blocks(
            (
                *envelope.runtime_instructions,
                *envelope.stage_contract,
                *envelope.agent_role_prompt,
                *envelope.task_objective,
                *envelope.specified_action,

Implement `ContextSizeGuard`:

```python
class ContextSizeGuard:
    def __init__(
        self,
        *,
        token_estimator: ContextTokenEstimator | None = None,
    ) -> None:
        self._token_estimator = token_estimator or ContextTokenEstimator()

    def compression_trigger_tokens(
        self,
        *,
        provider_snapshot: ProviderSnapshot,
        runtime_limit_snapshot: RuntimeLimitSnapshotRead,
        reserved_output_tokens: int = 0,
    ) -> int:
        base_threshold = floor(
            provider_snapshot.capabilities.context_window_tokens
            * runtime_limit_snapshot.context_limits.compression_threshold_ratio
        )
        if reserved_output_tokens <= 0:
            return base_threshold
        conservative_threshold = floor(
            max(1, provider_snapshot.capabilities.context_window_tokens - reserved_output_tokens)
            * runtime_limit_snapshot.context_limits.compression_threshold_ratio
        )
        return min(base_threshold, conservative_threshold)

    def apply_observation_budget(
        self,
        *,
        blocks: Sequence[ContextBlock],
        runtime_limit_snapshot: RuntimeLimitSnapshotRead,
        built_at: datetime,
    ) -> ObservationBudgetResult:
        raise NotImplementedError

    def apply_sliding_window(
        self,
        *,
        blocks: Sequence[ContextBlock],
        max_recent_blocks: int,
        stage_run_id: str,
        built_at: datetime,
    ) -> SlidingWindowResult:
        raise NotImplementedError

    def ensure_within_model_window(
        self,
        *,
        envelope: ContextEnvelope,
        provider_snapshot: ProviderSnapshot,
        runtime_limit_snapshot: RuntimeLimitSnapshotRead,
        reserved_output_tokens: int = 0,
    ) -> ContextSizeDecision:
        raise NotImplementedError
```

Implementation rules:

- Threshold formula uses only frozen `provider_snapshot.capabilities.context_window_tokens` and `runtime_limit_snapshot.context_limits.compression_threshold_ratio`.
- `reserved_output_tokens` only lowers the threshold.
- Pinned sections are `runtime_instructions`, `stage_contract`, `task_objective`, `response_schema`, `system_prompt` snapshot refs represented in `agent_role_prompt`, structured requirements and approved plan refs in `input_artifact_refs`, active file task refs, and available tool schemas.
- If pinned estimated tokens alone exceed the trigger threshold, raise `ContextOverflowError(reason="pinned_context_overflow", ...)`.
- If total estimated tokens reach the trigger threshold, return `requires_compression=True`; do not mutate run/stage state.
- Observation budgeting must preserve `content_ref` and source refs, set `truncated=True`, set `boundary_action=TRUNCATE`, and add a safe suffix such as `"... [truncated: 120 chars omitted; full_ref=artifact://...]"`.
- Sliding window must keep the latest `max_recent_blocks` blocks and replace older blocks with one `ContextBlock(section=original_section, boundary_action=REFERENCE_ONLY, content_ref="context-index://...")` that lists older refs in a bounded summary. It must not delete or claim to delete original process records.

def render_context_compression(
    self,
    request: PromptRenderRequest,
) -> PromptRenderResult:
    if request.model_call_type is not ModelCallType.CONTEXT_COMPRESSION:
        self._raise_unsupported_model_call_type(request)
    if not request.compression_source_context:
        raise PromptRenderException(
            PromptRenderError(
                code="compression_context_missing",
                message="compression_source_context is required for context compression.",
                stage_type=request.stage_type,
            )
        )
    asset = self._get_asset(COMPRESSION_PROMPT_ID)
    prompt_body = "\n\n".join(section.body for section in asset.sections)
    section = self._asset_section(
        request=request,
        section_id="compression_prompt",
        title="Context Compression",
        body="\n".join(
            [
                prompt_body,
                "Compression Trigger",
                request.compression_trigger_reason or "compression_threshold_exceeded",
                "Full trace ref:",
                request.full_trace_ref or "unavailable",
                "Response schema:",
                _stable_json(request.response_schema),
            ]
        ),
        prompt_ref=self._prompt_ref(asset),
        authority_level=PromptAuthorityLevel.SYSTEM_TRUSTED,
        cache_scope=asset.cache_scope,
    )
    messages = [
        PromptRenderedMessage(role="system", content=section.body),
        PromptRenderedMessage(role="user", content=request.compression_source_context),
    ]
    return self._result(
        request=request,
        messages=messages,
        sections=[section],
        model_call_type=ModelCallType.CONTEXT_COMPRESSION,
    )
```

@dataclass(frozen=True, slots=True)
class ContextCompressionRequest:
    envelope: ContextEnvelope
    manifest: ContextManifest
    stage_artifact_id: str
    trace_context: TraceContext
    full_trace_ref: str
    covered_step_range: str
    compression_trigger_reason: str
    provider_adapter: ProviderAdapterProtocol
    requested_max_output_tokens: int | None = None


class ContextCompressionRunner:
    def __init__(
        self,
        *,
        prompt_renderer: PromptRenderer,
        artifact_store: ArtifactStore,
        now: callable | None = None,
    ) -> None:
        self._prompt_renderer = prompt_renderer
        self._artifact_store = artifact_store
        self._now = now or (lambda: datetime.now(UTC))

    def compress(self, request: ContextCompressionRequest) -> ContextCompressionResult:
        raise NotImplementedError
```

Implementation rules:

- Build `PromptRenderRequest(model_call_type=ModelCallType.CONTEXT_COMPRESSION, compression_source_context=<bounded source text>, compression_trigger_reason=request.compression_trigger_reason, full_trace_ref=request.full_trace_ref)` from the current envelope and manifest.
- Use only `PromptRenderer.render_messages(...)` to obtain the compression prompt messages.
- Convert `PromptRenderedMessage(role="system"|"user")` into `SystemMessage` and `HumanMessage` for `provider_adapter.invoke_structured(...)`.
- Use `model_call_type=ModelCallType.CONTEXT_COMPRESSION`.
- Use `response_schema=COMPRESSED_CONTEXT_BLOCK_SCHEMA`.
- Do not pass tool descriptions for compression.
- Do not read compression prompt from environment variables, settings, template fields, or configuration packages.
- Validate `model_call_result.structured_output` into `CompressedContextBlock`; do not accept free text as the only compression output.
- Persist only through:
  - `append_process_record(process_key="compressed_context_block", ...)`
  - `append_process_record(process_key="context_compression_model_call_trace", ...)`
- Process values are list-preserving. The runner reads the existing process value through `ArtifactStore.get_stage_artifact(...)` and rewrites an accumulated list through `append_process_record(...)` so repeated compression attempts do not overwrite earlier evidence.
- Return `ContextCompressionResult(warning=...)` without a block when the provider returns an error or unparseable structured output; do not fabricate summary text.

def _block(
    *,
    block_id="block-1",
    section=ContextEnvelopeSection.WORKING_OBSERVATIONS,
    source_kind="bash_stdout",
    summary="short",
    content_ref="artifact://full-output-1",
    trust_level=ContextTrustLevel.UNTRUSTED_OBSERVATION,
    boundary_action=ContextBoundaryAction.REFERENCE_ONLY,
    estimated_tokens=None,
):
    return ContextBlock(
        block_id=block_id,
        section=section,
        trust_level=trust_level,
        boundary_action=boundary_action,
        summary=summary,
        content_ref=content_ref,
        sources=(
            ContextSourceRef(
                source_kind=source_kind,
                source_ref=content_ref,
                source_label=block_id,
            ),
        ),
        estimated_tokens=estimated_tokens,
        estimated_chars=len(summary),
    )
