# E3.1 Domain Event Store Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` to implement this plan task-by-task. Fallback to `superpowers:executing-plans` only if subagents cannot be dispatched or the task boundary cannot be represented safely. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the E3.1 domain event schema service and `EventStore` so run commands, Narrative Feed projections, and SSE increments share one event source.

**Architecture:** Keep the existing C1.9 `DomainEventModel` as the event.db persistence model, then add a service-layer domain event envelope, a PascalCase domain event mapping matrix, and append/list query APIs. `EventStore.append()` accepts a formal PascalCase domain event name, resolves the SSE `event_type`, validates the resulting payload through `SessionEvent`, persists the event model with `correlation_id`, and leaves transaction commit/rollback ownership to the caller.

**Tech Stack:** Python 3.11, SQLAlchemy 2.x ORM sessions, Pydantic v2 schemas, existing backend enums and feed/event schemas, pytest via `uv run python -m pytest`.

```python
def _coerce_domain_event_type(
    domain_event_type: DomainEventType | str,
) -> DomainEventType:
    if isinstance(domain_event_type, DomainEventType):
        return domain_event_type
    if domain_event_type in RAW_GRAPH_EVENT_TYPES:
        raise ValueError(f"{domain_event_type} is a raw graph event, not a domain event")
    try:
        return DomainEventType(domain_event_type)
    except ValueError as exc:
        raise ValueError(f"Unsupported domain event type: {domain_event_type}") from exc


def event_projection_rule(
    domain_event_type: DomainEventType | str,
) -> EventProjectionRule:
    event_type = _coerce_domain_event_type(domain_event_type)
    return EventProjectionMatrix[event_type]


def resolve_sse_event_type(domain_event_type: DomainEventType | str) -> SseEventType:
    return event_projection_rule(domain_event_type).sse_event_type


def resolve_feed_entry_type(
    domain_event_type: DomainEventType | str,
) -> FeedEntryType | None:
    return event_projection_rule(domain_event_type).feed_entry_type


def resolve_projection_targets(
    domain_event_type: DomainEventType | str,
) -> tuple[EventProjectionTarget, ...]:
    return event_projection_rule(domain_event_type).projection_targets
```

```python
@dataclass(frozen=True)
class DomainEvent:
    event_id: str
    session_id: str
    run_id: str | None
    event_type: SseEventType
    occurred_at: datetime
    payload: JsonObject
    stage_run_id: str | None
    sequence_index: int
    correlation_id: str | None
    causation_event_id: str | None

    @classmethod
    def from_model(cls, model: DomainEventModel) -> "DomainEvent":
        return cls(
            event_id=model.event_id,
            session_id=model.session_id,
            run_id=model.run_id,
            stage_run_id=model.stage_run_id,
            event_type=model.event_type,
            sequence_index=model.sequence_index,
            occurred_at=model.occurred_at,
            payload=dict(model.payload),
            correlation_id=model.correlation_id,
            causation_event_id=model.causation_event_id,
        )


class EventStore:
    def __init__(
        self,
        session: Session,
        *,
        now: Callable[[], datetime] | None = None,
        id_factory: Callable[[], str] | None = None,
    ) -> None:
        self._session = session
        self._now = now or (lambda: datetime.now(UTC))
        self._id_factory = id_factory or (lambda: f"event-{uuid4().hex}")

    def append(
        self,
        domain_event_type: DomainEventType | str,
        *,
        payload: JsonObject,
        trace_context: TraceContext,
        session_id: str | None = None,
        run_id: str | None = None,
        stage_run_id: str | None = None,
        event_id: str | None = None,
        occurred_at: datetime | None = None,
        causation_event_id: str | None = None,
    ) -> DomainEvent:
        resolved_session_id = session_id or trace_context.session_id
        if resolved_session_id is None:
            raise ValueError("session_id is required when TraceContext has no session_id")

        resolved_run_id = run_id if run_id is not None else trace_context.run_id
        resolved_stage_run_id = (
            stage_run_id if stage_run_id is not None else trace_context.stage_run_id
        )
        resolved_event_type = resolve_sse_event_type(domain_event_type)
        resolved_occurred_at = occurred_at or self._now()
        resolved_event_id = event_id or self._id_factory()

        SessionEvent(
            event_id=resolved_event_id,
            session_id=resolved_session_id,
            run_id=resolved_run_id,
            event_type=resolved_event_type,
            occurred_at=resolved_occurred_at,
            payload=payload,
        )

        model = DomainEventModel(
            event_id=resolved_event_id,
            session_id=resolved_session_id,
            run_id=resolved_run_id,
            stage_run_id=resolved_stage_run_id,
            event_type=resolved_event_type,
            sequence_index=self._next_sequence_index(resolved_session_id),
            occurred_at=resolved_occurred_at,
            payload=payload,
            correlation_id=trace_context.correlation_id,
            causation_event_id=causation_event_id,
            created_at=self._now(),
        )
        self._session.add(model)
        self._session.flush()
        return DomainEvent.from_model(model)

    def _next_sequence_index(self, session_id: str) -> int:
        current = self._session.execute(
            select(func.max(DomainEventModel.sequence_index)).where(
                DomainEventModel.session_id == session_id
            )
        ).scalar_one()
        return int(current or 0) + 1
```

```python
    def list_after(
        self,
        session_id: str,
        *,
        after_sequence_index: int,
        limit: int | None = None,
    ) -> list[DomainEvent]:
        statement = (
            select(DomainEventModel)
            .where(
                DomainEventModel.session_id == session_id,
                DomainEventModel.sequence_index > after_sequence_index,
            )
            .order_by(
                DomainEventModel.sequence_index.asc(),
                DomainEventModel.event_id.asc(),
            )
        )
        if limit is not None:
            statement = statement.limit(limit)
        return [
            DomainEvent.from_model(model)
            for model in self._session.execute(statement).scalars().all()
        ]

    def list_for_session(
        self,
        session_id: str,
        *,
        limit: int | None = None,
    ) -> list[DomainEvent]:
        statement = (
            select(DomainEventModel)
            .where(DomainEventModel.session_id == session_id)
            .order_by(
                DomainEventModel.sequence_index.asc(),
                DomainEventModel.event_id.asc(),
            )
        )
        if limit is not None:
            statement = statement.limit(limit)
        return [
            DomainEvent.from_model(model)
            for model in self._session.execute(statement).scalars().all()
        ]
```
